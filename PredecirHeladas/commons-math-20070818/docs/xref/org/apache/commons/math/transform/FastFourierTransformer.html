24  /***
25   * Implements the <a href="http://mathworld.wolfram.com/FastFourierTransform.html">
26   * Fast Fourier Transform</a> for transformation of one-dimensional data sets.
27   * For reference, see <b>Applied Numerical Linear Algebra</b>, ISBN 0898713897,
28   * chapter 6.
29   * <p>
30   * There are several conventions for the definition of FFT and inverse FFT,
31   * mainly on different coefficient and exponent. Here the equations are listed
32   * in the comments of the corresponding methods.
33   * <p>

35   * and speeds up the code. Users can pad the data with zeros to meet this
36   * requirement. There are other flavors of FFT, for reference, see S. Winograd,
37   * <i>On computing the discrete Fourier transform</i>, Mathematics of Computation,
38   * 32 (1978), 175 - 199.

288     /***
289      * Perform the base-4 Cooley-Tukey FFT algorithm (including inverse).
290      *
291      * @param f the real data array to be transformed
292      * @param isInverse the indicator of forward or inverse transform
293      * @return the complex transformed array
294      * @throws MathException if any math-related errors occur
295      * @throws IllegalArgumentException if any parameters are invalid
296      */
297     protected Complex[] fft(double f[], boolean isInverse) throws
298         MathException, IllegalArgumentException {
299 
300         verifyDataSet(f);
301         Complex F[] = new Complex[f.length];
302         if (f.length == 1) {
303             F[0] = new Complex(f[0], 0.0);
304             return F;
305         }
306 
307         // Rather than the naive real to complex conversion, pack 2N
308         // real numbers into N complex numbers for better performance.
309         int N = f.length >> 1;
310         Complex c[] = new Complex[N];
311         for (int i = 0; i < N; i++) {
312             c[i] = new Complex(f[2*i], f[2*i+1]);
313         }
314         computeOmega(isInverse ? -N : N);
315         Complex z[] = fft(c);
316 
317         // reconstruct the FFT result for the original array
318         computeOmega(isInverse ? -2*N : 2*N);
319         F[0] = new Complex(2 * (z[0].getReal() + z[0].getImaginary()), 0.0);
320         F[N] = new Complex(2 * (z[0].getReal() - z[0].getImaginary()), 0.0);
321         for (int i = 1; i < N; i++) {
322             Complex A = z[N-i].conjugate();
323             Complex B = z[i].add(A);
324             Complex C = z[i].subtract(A);
325             Complex D = omega[i].multiply(Complex.I);
326             F[i] = B.subtract(C.multiply(D));
327             F[2*N-i] = F[i].conjugate();
328         }
329 
330         return scaleArray(F, 0.5);
331     }

333     /***
334      * Perform the base-4 Cooley-Tukey FFT algorithm (including inverse).
335      *
336      * @param data the complex data array to be transformed
337      * @return the complex transformed array
338      * @throws MathException if any math-related errors occur
339      * @throws IllegalArgumentException if any parameters are invalid
340      */
341     protected Complex[] fft(Complex data[]) throws MathException,
342         IllegalArgumentException {
343 
344         int i, j, k, m, N = data.length;
345         Complex A, B, C, D, E, F, z, f[] = new Complex[N];
346 
347         // initial simple cases
348         verifyDataSet(data);
349         if (N == 1) {
350             f[0] = data[0];
351             return f;
352         }
353         if (N == 2) {
354             f[0] = data[0].add(data[1]);
355             f[1] = data[0].subtract(data[1]);
356             return f;
357         }
358 
359         // permute original data array in bit-reversal order
360         j = 0;
361         for (i = 0; i < N; i++) {
362             f[i] = data[j];
363             k = N >> 1;
364             while (j >= k && k > 0) {
365                 j -= k; k >>= 1;
366             }
367             j += k;
368         }
369 
370         // the bottom base-4 round
371         for (i = 0; i < N; i += 4) {
372             A = f[i].add(f[i+1]);
373             B = f[i+2].add(f[i+3]);
374             C = f[i].subtract(f[i+1]);
375             D = f[i+2].subtract(f[i+3]);
376             E = C.add(D.multiply(Complex.I));
377             F = C.subtract(D.multiply(Complex.I));
378             f[i] = A.add(B);
379             f[i+2] = A.subtract(B);
380             // omegaCount indicates forward or inverse transform
381             f[i+1] = omegaCount < 0 ? E : F;
382             f[i+3] = omegaCount > 0 ? E : F;
383         }
384 
385         // iterations from bottom to top take O(N*logN) time
386         for (i = 4; i < N; i <<= 1) {
387             m = N / (i<<1);
388             for (j = 0; j < N; j += i<<1) {
389                 for (k = 0; k < i; k++) {
390                     z = f[i+j+k].multiply(omega[k*m]);
391                     f[i+j+k] = f[j+k].subtract(z);
392                     f[j+k] = f[j+k].add(z);
393                 }
394             }
395         }
396         return f;
397     }

399     /***
400      * Calculate the n-th roots of unity.
401      * <p>
402      * The computed omega[] = { 1, w, w^2, ... w^(n-1) } where
403      * w = exp(-2 \pi i / n), i = sqrt(-1). Note n is positive for
404      * forward transform and negative for inverse transform.
405      * 
406      * @param n the integer passed in
407      * @throws IllegalArgumentException if n = 0
408      */
409     protected void computeOmega(int n) throws IllegalArgumentException {
410         if (n == 0) {
411             throw new IllegalArgumentException
412                 ("Cannot compute 0-th root of unity, indefinite result.");
413         }
414         // avoid repetitive calculations
415         if (n == omegaCount) { return; }
416         if (n + omegaCount == 0) {
417             for (int i = 0; i < Math.abs(omegaCount); i++) {
418                 omega[i] = omega[i].conjugate();
419             }
420             omegaCount = n;
421             return;
422         }
423         // calculate everything from scratch
424         omega = new Complex[Math.abs(n)];
425         double t = 2.0 * Math.PI / n;
426         double cost = Math.cos(t);
427         double sint = Math.sin(t);
428         omega[0] = new Complex(1.0, 0.0);
429         for (int i = 1; i < Math.abs(n); i++) {
430             omega[i] = new Complex(
431                 omega[i-1].getReal() * cost + omega[i-1].getImaginary() * sint,
432                 omega[i-1].getImaginary() * cost - omega[i-1].getReal() * sint);
433         }
434         omegaCount = n;
435     }

437     /***
438      * Sample the given univariate real function on the given interval.
439      * <p>
440      * The interval is divided equally into N sections and sample points
441      * are taken from min to max-(max-min)/N. Usually f(x) is periodic
442      * such that f(min) = f(max) (note max is not sampled), but we don't
443      * require that.
444      *
445      * @param f the function to be sampled
446      * @param min the lower bound for the interval
447      * @param max the upper bound for the interval
448      * @param n the number of sample points
449      * @return the samples array
450      * @throws MathException if any math-related errors occur
451      * @throws IllegalArgumentException if any parameters are invalid
452      */
453     public static double[] sample(
454         UnivariateRealFunction f, double min, double max, int n)
455         throws MathException, IllegalArgumentException {
456 
457         if (n <= 0) {
458             throw new IllegalArgumentException("Number of samples not positive.");
459         }
460         verifyInterval(min, max);
461 
462         double s[] = new double[n];
463         double h = (max - min) / n;
464         for (int i = 0; i < n; i++) {
465             s[i] = f.value(min + i * h);
466         }
467         return s;
468     }
