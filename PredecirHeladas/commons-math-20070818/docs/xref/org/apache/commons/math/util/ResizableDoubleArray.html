19  import java.io.Serializable;

21  /***
22   * <p>
23   * A variable length {@link DoubleArray} implementation that automatically 
24   * handles expanding and contracting its internal storage array as elements 
25   * are added and removed.
26   * </p>
27   * <p>
28   *  The internal storage array starts with capacity determined by the
29   * <code>initialCapacity</code> property, which can be set by the constructor.
30   * The default initial capacity is 16.  Adding elements using 
31   * {@link #addElement(double)} appends elements to the end of the array.  When 
32   * there are no open entries at the end of the internal storage array, the 
33   * array is expanded.  The size of the expanded array depends on the 
34   * <code>expansionMode</code> and <code>expansionFactor</code> properties.  
35   * The <code>expansionMode</code> determines whether the size of the array is 
36   * multiplied by the <code>expansionFactor</code> (MULTIPLICATIVE_MODE) or if 
37   * the expansion is additive (ADDITIVE_MODE -- <code>expansionFactor</code>
38   * storage locations added).  The default <code>expansionMode</code> is 
39   * MULTIPLICATIVE_MODE and the default <code>expansionFactor</code>
40   * is 2.0.
41   * </p>
42   * <p>
43   * The {@link #addElementRolling(double)} method adds a new element to the end
44   * of the internal storage array and adjusts the "usable window" of the 
45   * internal array forward by one position (effectively making what was the 
46   * second element the first, and so on).  Repeated activations of this method
47   * (or activation of {@link #discardFrontElements(int)}) will effectively orphan
48   * the storage locations at the beginning of the internal storage array.  To
49   * reclaim this storage, each time one of these methods is activated, the size
50   * of the internal storage array is compared to the number of addressable 
51   * elements (the <code>numElements</code> property) and if the difference
52   * is too large, the internal array is contracted to size 
53   * <code>numElements + 1.</code>  The determination of when the internal
54   * storage array is "too large" depends on the <code>expansionMode</code> and
55   * <code>contractionFactor</code> properties.  If  the <code>expansionMode</code>
56   * is <code>MULTIPLICATIVE_MODE</code>, contraction is triggered when the
57   * ratio between storage array length and <code>numElements</code> exceeds
58   * <code>contractionFactor.</code>  If the <code>expansionMode</code>
59   * is <code>ADDITIVE_MODE,</code> the number of excess storage locations
60   * is compared to <code>contractionFactor.</code>  
61   * </p>
62   * <p>
63   * To avoid cycles of expansions and contractions, the 
64   * <code>expansionFactor</code> must not exceed the 
65   * <code>contractionFactor.</code> Constructors and mutators for both of these
66   * properties enforce this requirement, throwing IllegalArgumentException if it
67   * is violated.
68   * </p>
69   * <p>
70   * @version $Revision: 480440 $ $Date: 2006-11-28 23:14:12 -0800 (Tue, 28 Nov 2006) $
71   */
72  public class ResizableDoubleArray implements DoubleArray, Serializable {

246     /***
247      * Adds an element to the end of this expandable array.
248      * 
249      * @param value to be added to end of array
250      */
251     public synchronized void addElement(double value) {
252         numElements++;
253         if ((startIndex + numElements) > internalArray.length) {
254             expand();
255         }
256         internalArray[startIndex + (numElements - 1)] = value;
257         if (shouldContract()) {
258             contract();
259         }
260     }
261 
262     /***
263      * <p>
264      * Adds an element to the end of the array and removes the first
265      * element in the array.  Returns the discarded first element.
266      * The effect is similar to a push operation in a FIFO queue.
267      * </p>
268      * <p>
269      * Example: If the array contains the elements 1, 2, 3, 4 (in that order)
270      * and addElementRolling(5) is invoked, the result is an array containing
271      * the entries 2, 3, 4, 5 and the value returned is 1.
272      * </p>
273      * 
274      * @param value the value to be added to the array
275      * @return the value which has been discarded or "pushed" out of the array
276      *         by this rolling insert
277      */
278     public synchronized double addElementRolling(double value) {
279         double discarded = internalArray[startIndex];
280 
281         if ((startIndex + (numElements + 1)) > internalArray.length) {
282             expand();
283         }
284         // Increment the start index
285         startIndex += 1;
286 
287         // Add the new value
288         internalArray[startIndex + (numElements - 1)] = value;
289 
290         // Check the contraction criteria
291         if (shouldContract()) {
292             contract();
293         }
294         return discarded;
295     }

297     /***
298      * Checks the expansion factor and the contraction criteria and throws an 
299      * IllegalArgumentException if the contractionCriteria is less than the 
300      * expansionCriteria
301      * 
302      * @param expansionFactor factor to be checked
303      * @param contractionCritera critera to be checked
304      * @throws IllegalArgumentException if the contractionCriteria is less than
305      *         the expansionCriteria.
306      */
307     protected void checkContractExpand(
308         float contractionCritera,
309         float expansionFactor) {
310 
311         if (contractionCritera < expansionFactor) {
312             String msg =
313                 "Contraction criteria can never be smaller than " +
314                 "the expansion factor.  This would lead to a never " +
315                 "ending loop of expansion and contraction as a newly " +
316                 "expanded internal storage array would immediately " +
317                 "satisfy the criteria for contraction";
318             throw new IllegalArgumentException(msg);
319         }
320 
321         if (contractionCriteria <= 1.0) {
322             String msg =
323                 "The contraction criteria must be a number larger " +
324                 "than one.  If the contractionCriteria is less than or " +
325                 "equal to one an endless loop of contraction and " +
326                 "expansion would ensue as an internalArray.length " +
327                 "== numElements would satisfy the contraction criteria";
328             throw new IllegalArgumentException(msg);
329         }
330 
331         if (expansionFactor <= 1.0) {
332             String msg =
333                 "The expansion factor must be a number greater than 1.0";
334             throw new IllegalArgumentException(msg);
335         }
336     }

342     public synchronized void clear() {
343         numElements = 0;
344         internalArray = new double[initialCapacity];
345     }

347     /***
348      * Contracts the storage array to the (size of the element set) + 1 - to 
349      * avoid a zero length array. This function also resets the startIndex to 
350      * zero. 
351      */
352     public synchronized void contract() {
353         double[] tempArray = new double[numElements + 1];
354 
355         // Copy and swap - copy only the element array from the src array.
356         System.arraycopy(internalArray, startIndex, tempArray, 0, numElements);
357         internalArray = tempArray;
358 
359         // Reset the start index to zero
360         startIndex = 0;
361     }

363     /***
364      * Discards the <code>i<code> initial elements of the array.  For example,
365      * if the array contains the elements 1,2,3,4, invoking 
366      * <code>discardFrontElements(2)</code> will cause the first two elements 
367      * to be discarded, leaving 3,4 in the array.  Throws illegalArgumentException
368      * if i exceeds numElements.
369      * 
370      * @param i  the number of elements to discard from the front of the array
371      * @throws IllegalArgumentException if i is greater than numElements.
372      */
373     public synchronized void discardFrontElements(int i) {
374         if (i > numElements) {
375             String msg = "Cannot discard more elements than are" +
376             "contained in this array.";
377             throw new IllegalArgumentException(msg);
378         } else if (i < 0) {
379             String msg = "Cannot discard a negative number of elements.";
380             throw new IllegalArgumentException(msg);
381         } else {
382             // "Subtract" this number of discarded from numElements 
383             numElements -= i;
384             startIndex += i;
385         }
386         if (shouldContract()) {
387             contract();
388         }
389     }

391     /***
392      * Expands the internal storage array using the expansion factor.
393      * <p>
394      * if <code>expansionMode</code> is set to MULTIPLICATIVE_MODE,
395      * the new array size will be <code>internalArray.length * expansionFactor.</code>
396      * If <code>expansionMode</code> is set to ADDITIVE_MODE,  the length
397      * after expansion will be <code>internalArray.length + expansionFactor</code>
398      */
399     protected synchronized void expand() {
400 
401         // notice the use of Math.ceil(), this gaurantees that we will always 
402         // have an array of at least currentSize + 1.   Assume that the 
403         // current initial capacity is 1 and the expansion factor
404         // is 1.000000000000000001.  The newly calculated size will be 
405         // rounded up to 2 after the multiplication is performed.
406         int newSize = 0;
407         if (expansionMode == MULTIPLICATIVE_MODE) {
408             newSize = (int) Math.ceil(internalArray.length * expansionFactor);
409         } else {
410             newSize = internalArray.length + Math.round(expansionFactor);
411         }
412         double[] tempArray = new double[newSize];
413 
414         // Copy and swap
415         System.arraycopy(internalArray, 0, tempArray, 0, internalArray.length);
416         internalArray = tempArray;
417     }
418     
419     /***
420      * Expands the internal storage array to the specified size.
421      * 
422      * @param size Size of the new internal storage array
423      */
424     private synchronized void expandTo(int size) {
425         double[] tempArray = new double[size];
426         // Copy and swap
427         System.arraycopy(internalArray, 0, tempArray, 0, internalArray.length);
428         internalArray = tempArray;
429     }

448      * Returns the element at the specified index
449      * 
450      * @param index index to fetch a value from
451      * @return value stored at the specified index
452      * @throws ArrayIndexOutOfBoundsException if <code>index</code> is less than

458                 "The index specified: " + index +
459                 " is larger than the current number of elements";
460             throw new ArrayIndexOutOfBoundsException(msg);
461         } else if (index >= 0) {
462             return internalArray[startIndex + index];
463         } else {
464             String msg =
465                 "Elements cannot be retrieved from a negative array index";
466             throw new ArrayIndexOutOfBoundsException(msg);
467         }
468     }

559     /***
560      * Sets the element at the specified index.  If the specified index is greater than
561      * <code>getNumElements() - 1</code>, the <code>numElements</code> property
562      * is increased to <code>index +1</code> and additional storage is allocated 
563      * (if necessary) for the new element and all  (uninitialized) elements 
564      * between the new element and the previous end of the array).
565      * 
566      * @param index index to store a value in
567      * @param value value to store at the specified index
568      * @throws ArrayIndexOutOfBoundsException if <code>index</code> is less than
569      *         zero.
570      */
571     public synchronized void setElement(int index, double value) {
572         if (index < 0) {
573             String msg = "Cannot set an element at a negative index";
574             throw new ArrayIndexOutOfBoundsException(msg);
575         }
576         if (index + 1 > numElements) {
577             numElements = index + 1;
578         }       
579         if ((startIndex + index) >= internalArray.length) {
580             expandTo(startIndex + (index + 1));
581         }    
582         internalArray[startIndex + index] = value;
583     }

592      * @param expansionFactor the new expansion factor value.
593      * @throws IllegalArgumentException if expansionFactor is <= 1 or greater
594      * than contractionFactor
595      */

597         checkContractExpand(getContractionCriteria(), expansionFactor);
598         // The check above verifies that the expansion factor is > 1.0;
599         this.expansionFactor = expansionFactor;
600     }

635     /***
636      * This function allows you to control the number of elements contained 
637      * in this array, and can be used to "throw out" the last n values in an 
638      * array. This function will also expand the internal array as needed.
639      * 
640      * @param i a new number of elements
641      * @throws IllegalArgumentException if <code>i</code> is negative.
642      */
643     public synchronized void setNumElements(int i) {
644 
645         // If index is negative thrown an error
646         if (i < 0) {
647             String msg =
648                 "Number of elements must be zero or a positive " + "integer";
649             throw new IllegalArgumentException(msg);
650         }
651 
652         // Test the new num elements, check to see if the array needs to be 
653         // expanded to accomodate this new number of elements
654         if ((startIndex + i) > internalArray.length) {
655             expandTo(startIndex + i);
656         }
657 
658         // Set the new number of elements to new value
659         numElements = i;
660     }

662     /***
663      * Returns true if the internal storage array has too many unused 
664      * storage positions.  
665      * 
666      * @return true if array satisfies the contraction criteria
667      */
668     private synchronized boolean shouldContract() {
669         if (expansionMode == MULTIPLICATIVE_MODE) {
