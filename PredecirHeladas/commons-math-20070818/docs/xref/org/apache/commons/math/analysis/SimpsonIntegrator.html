17  package org.apache.commons.math.analysis;
18  
19  import org.apache.commons.math.FunctionEvaluationException;
20  import org.apache.commons.math.MaxIterationsExceededException;
21  
22  /***
23   * Implements the <a href="http://mathworld.wolfram.com/SimpsonsRule.html">
24   * Simpson's Rule</a> for integration of real univariate functions. For
25   * reference, see <b>Introduction to Numerical Analysis</b>, ISBN 038795452X,
26   * chapter 3.
27   * <p>
28   * This implementation employs basic trapezoid rule as building blocks to
29   * calculate the Simpson's rule of alternating 2/3 and 4/3.
30   *  
31   * @version $Revision: 506585 $ $Date: 2007-02-12 11:17:41 -0800 (Mon, 12 Feb 2007) $
32   */
33  public class SimpsonIntegrator extends UnivariateRealIntegratorImpl {
34  
35      /*** serializable version identifier */
36      private static final long serialVersionUID = 3405465123320678216L;
37  
38      /***
39       * Construct an integrator for the given function.
40       * 
41       * @param f function to integrate
42       */
43      public SimpsonIntegrator(UnivariateRealFunction f) {
44          super(f, 64);
45      }
46  
47      /***
48       * Integrate the function in the given interval.
49       * 
50       * @param min the lower bound for the interval
51       * @param max the upper bound for the interval
52       * @return the value of integral
53       * @throws MaxIterationsExceededException if the maximum iteration count is exceeded
54       * or the integrator detects convergence problems otherwise
55       * @throws FunctionEvaluationException if an error occurs evaluating the
56       * function
57       * @throws IllegalArgumentException if any parameters are invalid
58       */
59      public double integrate(double min, double max) throws MaxIterationsExceededException,
60          FunctionEvaluationException, IllegalArgumentException {
61          
62          int i = 1;
63          double s, olds, t, oldt;
64          
65          clearResult();
66          verifyInterval(min, max);
67          verifyIterationCount();
68  
69          TrapezoidIntegrator qtrap = new TrapezoidIntegrator(this.f);
70          if (minimalIterationCount == 1) {
71              s = (4 * qtrap.stage(min, max, 1) - qtrap.stage(min, max, 0)) / 3.0;
72              setResult(s, 1);
73              return result;
74          }
75          // Simpson's rule requires at least two trapezoid stages.
76          olds = 0;
77          oldt = qtrap.stage(min, max, 0);
78          while (i <= maximalIterationCount) {
79              t = qtrap.stage(min, max, i);
80              s = (4 * t - oldt) / 3.0;
81              if (i >= minimalIterationCount) {
82                  if (Math.abs(s - olds) <= Math.abs(relativeAccuracy * olds)) {
83                      setResult(s, i);
84                      return result;
85                  }
86              }
87              olds = s;
88              oldt = t;
89              i++;
90          }
91          throw new MaxIterationsExceededException(maximalIterationCount);
92      }
93  
94      /***
95       * Verifies that the iteration limits are valid and within the range.
96       * 
97       * @throws IllegalArgumentException if not
98       */
99      protected void verifyIterationCount() throws IllegalArgumentException {
100         super.verifyIterationCount();
101         // at most 64 bisection refinements
102         if (maximalIterationCount > 64) {
103             throw new IllegalArgumentException
104                 ("Iteration upper limit out of [0, 64] range: " +
105                 maximalIterationCount);
106         }
107     }
108 }
