33  /***
34   * Implements <code>EmpiricalDistribution</code> interface.  This implementation
35   * uses what amounts to the
36   * <a href="http://nedwww.ipac.caltech.edu/level5/March02/Silverman/Silver2_6.html">
37   * Variable Kernel Method</a> with Gaussian smoothing:<p>
38   * <strong>Digesting the input file</strong>
39   * <ol><li>Pass the file once to compute min and max.</li>
40   * <li>Divide the range from min-max into <code>binCount</code> "bins."</li>
41   * <li>Pass the data file again, computing bin counts and univariate
42   *     statistics (mean, std dev.) for each of the bins </li>
43   * <li>Divide the interval (0,1) into subintervals associated with the bins,
44   *     with the length of a bin's subinterval proportional to its count.</li></ol>
45   * <strong>Generating random values from the distribution</strong><ol>
46   * <li>Generate a uniformly distributed value in (0,1) </li>
47   * <li>Select the subinterval to which the value belongs.
48   * <li>Generate a random Gaussian value with mean = mean of the associated
49   *     bin and std dev = std dev of associated bin.</li></ol></p><p>
50   *<strong>USAGE NOTES:</strong><ul>
51   *<li>The <code>binCount</code> is set by default to 1000.  A good rule of thumb
52   *    is to set the bin count to approximately the length of the input file divided
53   *    by 10. </li>
54   *<li>The input file <i>must</i> be a plain text file containing one valid numeric
55   *    entry per line.</li>
56   * </ul></p>
57   *
58   * @version $Revision: 480440 $ $Date: 2006-11-28 23:14:12 -0800 (Tue, 28 Nov 2006) $
59   */

177     /***
178      * Provides methods for computing <code>sampleStats</code> and
179      * <code>beanStats</code> abstracting the source of data.
180      */
181     private abstract class DataAdapter{
182         /*** 
183          * Compute bin stats.
184          * 
185          * @param min minimum value
186          * @param delta  grid size
187          * @throws Exception  if an error occurs computing bin stats
188          */
189         public abstract void computeBinStats(double min, double delta)
190                 throws Exception;
191         /***
192          * Compute sample statistics.
193          * 
194          * @throws Exception if an error occurs computing sample stats
195          */
196         public abstract void computeStats() throws Exception;
197     }
198     /***
199      * Factory of <code>DataAdapter</code> objects. For every supported source
200      * of data (array of doubles, file, etc.) an instance of the proper object
201      * is returned.
202      */
203     private class DataAdapterFactory{
204         /***
205          * Creates a DataAdapter from a data object
206          * 
207          * @param in object providing access to the data
208          * @return DataAdapter instance
209          */
210         public DataAdapter getAdapter(Object in) {
211             if (in instanceof BufferedReader) {
212                 BufferedReader inputStream = (BufferedReader) in;
213                 return new StreamDataAdapter(inputStream);
214             } else if (in instanceof double[]) {
215                 double[] inputArray = (double[]) in;
216                 return new ArrayDataAdapter(inputArray);
217             } else {
218                 throw new IllegalArgumentException(
219                     "Input data comes from the" + " unsupported source");
220             }
221         }
222     }
223     /***
224      * <code>DataAdapter</code> for data provided through some input stream
225      */
226     private class StreamDataAdapter extends DataAdapter{
227         
228         /*** Input stream providng access to the data */
229         private BufferedReader inputStream;
230         
231         /***
232          * Create a StreamDataAdapter from a BufferedReader
233          * 
234          * @param in BufferedReader input stream
235          */
236         public StreamDataAdapter(BufferedReader in){
237             super();
238             inputStream = in;
239         }
240         /***
241          * Computes binStats
242          * 
243          * @param min  minimum value
244          * @param delta  grid size
245          * @throws IOException if an IO error occurs
246          */
247         public void computeBinStats(double min, double delta)
248                 throws IOException {
249             String str = null;
250             double val = 0.0d;
251             while ((str = inputStream.readLine()) != null) {
252                 val = Double.parseDouble(str);
253                 SummaryStatistics stats =
254                     (SummaryStatistics) binStats.get(findBin(min, val, delta));
255                 stats.addValue(val);
256             }
257 
258             inputStream.close();
259             inputStream = null;
260         }
261         /***
262          * Computes sampleStats
263          * 
264          * @throws IOException if an IOError occurs
265          */
266         public void computeStats() throws IOException {
267             String str = null;
268             double val = 0.0;
269             sampleStats = SummaryStatistics.newInstance();
270             while ((str = inputStream.readLine()) != null) {
271                 val = new Double(str).doubleValue();
272                 sampleStats.addValue(val);
273             }
274             inputStream.close();
275             inputStream = null;
276         }
277     }

279     /***
280      * <code>DataAdapter</code> for data provided as array of doubles.
281      */
282     private class ArrayDataAdapter extends DataAdapter{
283         
284         /*** Array of input  data values */
285         private double[] inputArray;
286         
287         /***
288          * Construct an ArrayDataAdapter from a double[] array
289          * 
290          * @param in double[] array holding the data
291          */
292         public ArrayDataAdapter(double[] in){
293             super();
294             inputArray = in;
295         }
296         /***
297          * Computes sampleStats
298          * 
299          * @throws IOException if an IO error occurs
300          */
301         public void computeStats() throws IOException {
302             sampleStats = SummaryStatistics.newInstance();
303             for (int i = 0; i < inputArray.length; i++) {
304                 sampleStats.addValue(inputArray[i]);
305             }
306         }
307         /***
308          * Computes binStats
309          * 
310          * @param min  minimum value
311          * @param delta  grid size
312          * @throws IOException  if an IO error occurs
313          */
314         public void computeBinStats(double min, double delta)
315             throws IOException {
316             for (int i = 0; i < inputArray.length; i++) {
317                 SummaryStatistics stats =
318                     (SummaryStatistics) binStats.get(
319                             findBin(min, inputArray[i], delta));
320                 stats.addValue(inputArray[i]);
321             }
322         }
323     }

325     /***
326      * Fills binStats array (second pass through data file).
327      * 
328      * @param in object providing access to the data
329      * @throws IOException  if an IO error occurs
330      */
331     private void fillBinStats(Object in) throws IOException {
332         // Load array of bin upper bounds -- evenly spaced from min - max
333         double min = sampleStats.getMin();
334         double max = sampleStats.getMax();
335         double delta = (max - min)/(new Double(binCount)).doubleValue();
336         double[] binUpperBounds = new double[binCount];
337         binUpperBounds[0] = min + delta;
338         for (int i = 1; i< binCount - 1; i++) {
339             binUpperBounds[i] = binUpperBounds[i-1] + delta;
340         }
341         binUpperBounds[binCount -1] = max;
342 
343         // Initialize binStats ArrayList
344         if (!binStats.isEmpty()) {
345             binStats.clear();
346         }
347         for (int i = 0; i < binCount; i++) {
348             SummaryStatistics stats = SummaryStatistics.newInstance();
349             binStats.add(i,stats);
350         }
351 
352         // Filling data in binStats Array
353         DataAdapterFactory aFactory = new DataAdapterFactory();
354         DataAdapter da = aFactory.getAdapter(in);
355         try {
356             da.computeBinStats(min, delta);
357         } catch (Exception e) {
358             if(e instanceof RuntimeException){
359                 throw new RuntimeException(e.getMessage());
360             }else{
361                 throw new IOException(e.getMessage());
362             }
363         }
364 
365         // Assign upperBounds based on bin counts
366         upperBounds = new double[binCount];
367         upperBounds[0] =
368         ((double)((SummaryStatistics)binStats.get(0)).getN())/
369         (double)sampleStats.getN();
370         for (int i = 1; i < binCount-1; i++) {
371             upperBounds[i] = upperBounds[i-1] +
372             ((double)((SummaryStatistics)binStats.get(i)).getN())/
373             (double)sampleStats.getN();
374         }
375         upperBounds[binCount-1] = 1.0d;
376     }

378     /***
379      * Returns the index of the bin to which the given value belongs
380      * 
381      * @param min  the minimum value
382      * @param value  the value whose bin we are trying to find
383      * @param delta  the grid size
384      * @return the index of the bin containing the value
385      */
386     private int findBin(double min, double value, double delta) {
387         return Math.min(
388                 Math.max((int) Math.ceil((value- min) / delta) - 1, 0), 
389                 binCount - 1);
390         }

392     /***
393      * Generates a random value from this distribution.
394      * 
395      * @return the random value.
396      * @throws IllegalStateException if the distribution has not been loaded
397      */
398     public double getNextValue() throws IllegalStateException {
399 
400         if (!loaded) {
401             throw new IllegalStateException("distribution not loaded");
402         }
403 
404         // Start with a uniformly distributed random number in (0,1)
405         double x = Math.random();
406 
407         // Use this to select the bin and generate a Gaussian within the bin
408         for (int i = 0; i < binCount; i++) {
409            if (x <= upperBounds[i]) {
410                SummaryStatistics stats = (SummaryStatistics)binStats.get(i);
411                if (stats.getN() > 0) {
412                    if (stats.getStandardDeviation() > 0) {  // more than one obs
413                         return randomData.nextGaussian
414                             (stats.getMean(),stats.getStandardDeviation());
415                    } else {
416                        return stats.getMean(); // only one obs in bin
417                    }
418                }
419            }
420         }
421         throw new RuntimeException("No bin selected");
422     }
