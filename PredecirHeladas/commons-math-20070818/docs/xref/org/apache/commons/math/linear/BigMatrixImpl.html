18  package org.apache.commons.math.linear;
19  import java.io.Serializable;
20  import java.math.BigDecimal;

22  /***
23   * Implementation of {@link BigMatrix} using a BigDecimal[][] array to store entries
24   * and <a href="http://www.math.gatech.edu/~bourbaki/math2601/Web-notes/2num.pdf">
25   * LU decompostion</a> to support linear system 
26   * solution and inverse.
27   * <p>
28   * The LU decompostion is performed as needed, to support the following operations: <ul>
29   * <li>solve</li>
30   * <li>isSingular</li>
31   * <li>getDeterminant</li>
32   * <li>inverse</li> </ul>
33   * <p>
34  * <strong>Usage notes</strong>:<br>
35   * <ul><li>
36   * The LU decomposition is stored and reused on subsequent calls.  If matrix
37   * data are modified using any of the public setXxx methods, the saved
38   * decomposition is discarded.  If data are modified via references to the
39   * underlying array obtained using <code>getDataRef()</code>, then the stored
40   * LU decomposition will not be discarded.  In this case, you need to
41   * explicitly invoke <code>LUDecompose()</code> to recompute the decomposition
42   * before using any of the methods above.</li>
43   * <li>
44   * As specified in the {@link BigMatrix} interface, matrix element indexing
45   * is 0-based -- e.g., <code>getEntry(0, 0)</code>
46   * returns the element in the first row, first column of the matrix.</li></ul>
47   * @version $Revision: 480440 $ $Date: 2006-11-28 23:14:12 -0800 (Tue, 28 Nov 2006) $
48   */
49  public class BigMatrixImpl implements BigMatrix, Serializable {

121     /***
122      * Create a new BigMatrix using the <code>data</code> as the underlying
123      * data array.
124      * <p>
125      * The input array is copied, not referenced.
126      *
127      * @param d data for new matrix
128      * @throws IllegalArgumentException if <code>d</code> is not rectangular
129      *  (not all rows have the same length) or empty
130      * @throws NullPointerException if <code>d</code> is null
131      */
132     public BigMatrixImpl(double[][] d) {
133         int nRows = d.length;
134         if (nRows == 0) {
135             throw new IllegalArgumentException(
136             "Matrix must have at least one row."); 
137         }
138         int nCols = d[0].length;
139         if (nCols == 0) {
140             throw new IllegalArgumentException(
141             "Matrix must have at least one column."); 
142         }
143         for (int row = 1; row < nRows; row++) {
144             if (d[row].length != nCols) {
145                 throw new IllegalArgumentException(
146                 "All input rows must have the same length.");
147             }
148         }
149         this.copyIn(d);
150         lu = null;
151     }

209     /***
210      * Compute the sum of this and <code>m</code>.
211      *
212      * @param m    matrix to be added
213      * @return     this + m
214      * @exception  IllegalArgumentException if m is not the same size as this
215      */
216     public BigMatrix add(BigMatrix m) throws IllegalArgumentException {
217         if (this.getColumnDimension() != m.getColumnDimension() ||
218                 this.getRowDimension() != m.getRowDimension()) {
219             throw new IllegalArgumentException("matrix dimension mismatch");
220         }
221         int rowCount = this.getRowDimension();
222         int columnCount = this.getColumnDimension();
223         BigDecimal[][] outData = new BigDecimal[rowCount][columnCount];
224         for (int row = 0; row < rowCount; row++) {
225             for (int col = 0; col < columnCount; col++) {
226                 outData[row][col] = data[row][col].add(m.getEntry(row, col));
227             }
228         }
229         return new BigMatrixImpl(outData);
230     }

232     /***
233      * Compute  this minus <code>m</code>.
234      *
235      * @param m    matrix to be subtracted
236      * @return     this + m
237      * @exception  IllegalArgumentException if m is not the same size as *this
238      */
239     public BigMatrix subtract(BigMatrix m) throws IllegalArgumentException {
240         if (this.getColumnDimension() != m.getColumnDimension() ||
241                 this.getRowDimension() != m.getRowDimension()) {
242             throw new IllegalArgumentException("matrix dimension mismatch");
243         }
244         int rowCount = this.getRowDimension();
245         int columnCount = this.getColumnDimension();
246         BigDecimal[][] outData = new BigDecimal[rowCount][columnCount];
247         for (int row = 0; row < rowCount; row++) {
248             for (int col = 0; col < columnCount; col++) {
249                 outData[row][col] = data[row][col].subtract(m.getEntry(row, col));
250             }
251         }
252         return new BigMatrixImpl(outData);
253     }

290     /***
291      * Returns the result of postmultiplying this by <code>m</code>.
292      * @param m    matrix to postmultiply by
293      * @return     this*m
294      * @throws     IllegalArgumentException
295      *             if columnDimension(this) != rowDimension(m)
296      */
297     public BigMatrix multiply(BigMatrix m) throws IllegalArgumentException {
298         if (this.getColumnDimension() != m.getRowDimension()) {
299             throw new IllegalArgumentException("Matrices are not multiplication compatible.");
300         }
301         int nRows = this.getRowDimension();
302         int nCols = m.getColumnDimension();
303         int nSum = this.getColumnDimension();
304         BigDecimal[][] outData = new BigDecimal[nRows][nCols];
305         BigDecimal sum = ZERO;
306         for (int row = 0; row < nRows; row++) {
307             for (int col = 0; col < nCols; col++) {
308                 sum = ZERO;
309                 for (int i = 0; i < nSum; i++) {
310                     sum = sum.add(data[row][i].multiply(m.getEntry(i, col)));
311                 }
312                 outData[row][col] = sum;
313             }
314         }
315         return new BigMatrixImpl(outData);
316     }

409     /***
410      * Returns the <a href="http://mathworld.wolfram.com/MaximumAbsoluteRowSumNorm.html">
411      * maximum absolute row sum norm</a> of the matrix.
412      *
413      * @return norm
414      */
415     public BigDecimal getNorm() {
416         BigDecimal maxColSum = ZERO;
417         for (int col = 0; col < this.getColumnDimension(); col++) {
418             BigDecimal sum = ZERO;
419             for (int row = 0; row < this.getRowDimension(); row++) {
420                 sum = sum.add(data[row][col].abs());
421             }
422             maxColSum = maxColSum.max(sum);
423         }
424         return maxColSum;
425     }

427     /***
428      * Gets a submatrix. Rows and columns are indicated
429      * counting from 0 to n-1.
430      *
431      * @param startRow Initial row index
432      * @param endRow Final row index
433      * @param startColumn Initial column index
434      * @param endColumn Final column index
435      * @return The subMatrix containing the data of the
436      *         specified rows and columns
437      * @exception MatrixIndexException if row or column selections are not valid
438      */
439     public BigMatrix getSubMatrix(int startRow, int endRow, int startColumn,
440             int endColumn) throws MatrixIndexException {
441         if (startRow < 0 || startRow > endRow || endRow > data.length ||
442                 startColumn < 0 || startColumn > endColumn ||
443                 endColumn > data[0].length ) {
444             throw new MatrixIndexException(
445             "invalid row or column index selection");
446         }
447         BigMatrixImpl subMatrix = new BigMatrixImpl(endRow - startRow+1,
448                 endColumn - startColumn+1);
449         BigDecimal[][] subMatrixData = subMatrix.getDataRef();
450         for (int i = startRow; i <= endRow; i++) {
451             for (int j = startColumn; j <= endColumn; j++) {
452                 subMatrixData[i - startRow][j - startColumn] = data[i][j];
453             }
454         }
455         return subMatrix;
456     }

458     /***
459      * Gets a submatrix. Rows and columns are indicated
460      * counting from 0 to n-1.
461      *
462      * @param selectedRows Array of row indices must be non-empty
463      * @param selectedColumns Array of column indices must be non-empty
464      * @return The subMatrix containing the data in the
465      *     specified rows and columns
466      * @exception MatrixIndexException  if supplied row or column index arrays
467      *     are not valid
468      */
469     public BigMatrix getSubMatrix(int[] selectedRows, int[] selectedColumns)
470     throws MatrixIndexException {
471         if (selectedRows.length * selectedColumns.length == 0) {
472             throw new MatrixIndexException(
473             "selected row and column index arrays must be non-empty");
474         }
475         BigMatrixImpl subMatrix = new BigMatrixImpl(selectedRows.length,
476                 selectedColumns.length);
477         BigDecimal[][] subMatrixData = subMatrix.getDataRef();
478         try  {
479             for (int i = 0; i < selectedRows.length; i++) {
480                 for (int j = 0; j < selectedColumns.length; j++) {
481                     subMatrixData[i][j] = data[selectedRows[i]][selectedColumns[j]];
482                 }
483             }
484         }
485         catch (ArrayIndexOutOfBoundsException e) {
486             throw new MatrixIndexException("matrix dimension mismatch");
487         }
488         return subMatrix;
489     }

491     /***
492      * Replace the submatrix starting at <code>row, column</code> using data in
493      * the input <code>subMatrix</code> array. Indexes are 0-based.
494      * <p> 
495      * Example:<br>
496      * Starting with <pre>
497      * 1  2  3  4
498      * 5  6  7  8
499      * 9  0  1  2
500      * </pre>
501      * and <code>subMatrix = {{3, 4} {5,6}}</code>, invoking 
502      * <code>setSubMatrix(subMatrix,1,1))</code> will result in <pre>
503      * 1  2  3  4
504      * 5  3  4  8
505      * 9  5  6  2
506      * </pre>
507      * 
508      * @param subMatrix  array containing the submatrix replacement data
509      * @param row  row coordinate of the top, left element to be replaced
510      * @param column  column coordinate of the top, left element to be replaced
511      * @throws MatrixIndexException  if subMatrix does not fit into this 
512      *    matrix from element in (row, column) 
513      * @throws IllegalArgumentException if <code>subMatrix</code> is not rectangular
514      *  (not all rows have the same length) or empty
515      * @throws NullPointerException if <code>subMatrix</code> is null
516      * @since 1.1
517      */
518     public void setSubMatrix(BigDecimal[][] subMatrix, int row, int column) 
519     throws MatrixIndexException {
520         if ((row < 0) || (column < 0)){
521             throw new MatrixIndexException
522             ("invalid row or column index selection");          
523         }
524         int nRows = subMatrix.length;
525         if (nRows == 0) {
526             throw new IllegalArgumentException(
527             "Matrix must have at least one row."); 
528         }
529         int nCols = subMatrix[0].length;
530         if (nCols == 0) {
531             throw new IllegalArgumentException(
532             "Matrix must have at least one column."); 
533         }
534         for (int r = 1; r < nRows; r++) {
535             if (subMatrix[r].length != nCols) {
536                 throw new IllegalArgumentException(
537                 "All input rows must have the same length.");
538             }
539         }       
540         if (data == null) {
541             if ((row > 0)||(column > 0)) throw new MatrixIndexException
542             ("matrix must be initialized to perfom this method");
543             data = new BigDecimal[nRows][nCols];
544             System.arraycopy(subMatrix, 0, data, 0, subMatrix.length);          
545         }   
546         if (((nRows + row) > this.getRowDimension()) ||
547             (nCols + column > this.getColumnDimension()))
548             throw new MatrixIndexException(
549             "invalid row or column index selection");                   
550         for (int i = 0; i < nRows; i++) {
551             System.arraycopy(subMatrix[i], 0, data[row + i], column, nCols);
552         } 
553         lu = null;
554     }

944     /***
945      * Returns a matrix of (column) solution vectors for linear systems with
946      * coefficient matrix = this and constant vectors = columns of
947      * <code>b</code>. 
948      *
949      * @param b  matrix of constant vectors forming RHS of linear systems to
950      * to solve
951      * @return matrix of solution vectors
952      * @throws IllegalArgumentException if this.rowDimension != row dimension
953      * @throws InvalidMatrixException if this matrix is not square or is singular
954      */
955     public BigMatrix solve(BigMatrix b) throws IllegalArgumentException, InvalidMatrixException  {
956         if (b.getRowDimension() != this.getRowDimension()) {
957             throw new IllegalArgumentException("Incorrect row dimension");
958         }
959         if (!this.isSquare()) {
960             throw new InvalidMatrixException("coefficient matrix is not square");
961         }
962         if (this.isSingular()) { // side effect: compute LU decomp
963             throw new InvalidMatrixException("Matrix is singular.");
964         }
965         
966         int nCol = this.getColumnDimension();
967         int nColB = b.getColumnDimension();
968         int nRowB = b.getRowDimension();
969         
970         // Apply permutations to b
971         BigDecimal[][] bp = new BigDecimal[nRowB][nColB];
972         for (int row = 0; row < nRowB; row++) {
973             for (int col = 0; col < nColB; col++) {
974                 bp[row][col] = b.getEntry(permutation[row], col);
975             }
976         }
977         
978         // Solve LY = b
979         for (int col = 0; col < nCol; col++) {
980             for (int i = col + 1; i < nCol; i++) {
981                 for (int j = 0; j < nColB; j++) {
982                     bp[i][j] = bp[i][j].subtract(bp[col][j].multiply(lu[i][col]));
983                 }
984             }
985         }
986         
987         // Solve UX = Y
988         for (int col = nCol - 1; col >= 0; col--) {
989             for (int j = 0; j < nColB; j++) {
990                 bp[col][j] = bp[col][j].divide(lu[col][col], scale, roundingMode);
991             }
992             for (int i = 0; i < col; i++) {
993                 for (int j = 0; j < nColB; j++) {
994                     bp[i][j] = bp[i][j].subtract(bp[col][j].multiply(lu[i][col]));
995                 }
996             }
997         }
998         
999         BigMatrixImpl outMat = new BigMatrixImpl(bp);
1000         return outMat;
1001     }

1003     /***
1004      * Computes a new 
1005      * <a href="http://www.math.gatech.edu/~bourbaki/math2601/Web-notes/2num.pdf">
1006      * LU decompostion</a> for this matrix, storing the result for use by other methods. 
1007      * <p>
1008      * <strong>Implementation Note</strong>:<br>
1009      * Uses <a href="http://www.damtp.cam.ac.uk/user/fdl/people/sd/lectures/nummeth98/linear.htm">
1010      * Crout's algortithm</a>, with partial pivoting.
1011      * <p>
1012      * <strong>Usage Note</strong>:<br>
1013      * This method should rarely be invoked directly. Its only use is
1014      * to force recomputation of the LU decomposition when changes have been
1015      * made to the underlying data using direct array references. Changes
1016      * made using setXxx methods will trigger recomputation when needed
1017      * automatically.
1018      *
1019      * @throws InvalidMatrixException if the matrix is non-square or singular.
1020      */
1021     public void luDecompose() throws InvalidMatrixException {
1022         
1023         int nRows = this.getRowDimension();
1024         int nCols = this.getColumnDimension();
1025         if (nRows != nCols) {
1026             throw new InvalidMatrixException("LU decomposition requires that the matrix be square.");
1027         }
1028         lu = this.getData();
1029         
1030         // Initialize permutation array and parity
1031         permutation = new int[nRows];
1032         for (int row = 0; row < nRows; row++) {
1033             permutation[row] = row;
1034         }
1035         parity = 1;
1036         
1037         // Loop over columns
1038         for (int col = 0; col < nCols; col++) {
1039             
1040             BigDecimal sum = ZERO;
1041             
1042             // upper
1043             for (int row = 0; row < col; row++) {
1044                 sum = lu[row][col];
1045                 for (int i = 0; i < row; i++) {
1046                     sum = sum.subtract(lu[row][i].multiply(lu[i][col]));
1047                 }
1048                 lu[row][col] = sum;
1049             }
1050             
1051             // lower
1052             int max = col; // permutation row
1053             BigDecimal largest = ZERO;
1054             for (int row = col; row < nRows; row++) {
1055                 sum = lu[row][col];
1056                 for (int i = 0; i < col; i++) {
1057                     sum = sum.subtract(lu[row][i].multiply(lu[i][col]));
1058                 }
1059                 lu[row][col] = sum;
1060                 
1061                 // maintain best permutation choice
1062                 if (sum.abs().compareTo(largest) == 1) {
1063                     largest = sum.abs();
1064                     max = row;
1065                 }
1066             }
1067             
1068             // Singularity check
1069             if (lu[max][col].abs().compareTo(TOO_SMALL) <= 0) {
1070                 lu = null;
1071                 throw new InvalidMatrixException("matrix is singular");
1072             }
1073             
1074             // Pivot if necessary
1075             if (max != col) {
1076                 BigDecimal tmp = ZERO;
1077                 for (int i = 0; i < nCols; i++) {
1078                     tmp = lu[max][i];
1079                     lu[max][i] = lu[col][i];
1080                     lu[col][i] = tmp;
1081                 }
1082                 int temp = permutation[max];
1083                 permutation[max] = permutation[col];
1084                 permutation[col] = temp;
1085                 parity = -parity;
1086             }
1087             
1088             //Divide the lower elements by the "winning" diagonal elt.
1089             for (int row = col + 1; row < nRows; row++) {
1090                 lu[row][col] = lu[row][col].divide(lu[col][col], scale, roundingMode);
1091             }
1092             
1093         }
1094         
1095     }
