24  /***
25   * Implementation of RealMatrix using a double[][] array to store entries and
26   * <a href="http://www.math.gatech.edu/~bourbaki/math2601/Web-notes/2num.pdf">
27   * LU decompostion</a> to support linear system
28   * solution and inverse.
29   * <p>
30   * The LU decompostion is performed as needed, to support the following operations: <ul>
31   * <li>solve</li>
32   * <li>isSingular</li>
33   * <li>getDeterminant</li>
34   * <li>inverse</li> </ul>
35   * <p>
36   * <strong>Usage notes</strong>:<br>
37   * <ul><li>
38   * The LU decomposition is cached and reused on subsequent calls.   
39   * If data are modified via references to the underlying array obtained using
40   * <code>getDataRef()</code>, then the stored LU decomposition will not be
41   * discarded.  In this case, you need to explicitly invoke 
42   * <code>LUDecompose()</code> to recompute the decomposition
43   * before using any of the methods above.</li>
44   * <li>
45   * As specified in the {@link RealMatrix} interface, matrix element indexing
46   * is 0-based -- e.g., <code>getEntry(0, 0)</code>
47   * returns the element in the first row, first column of the matrix.</li></ul>
48   *
49   * @version $Revision: 480440 $ $Date: 2006-11-28 23:14:12 -0800 (Tue, 28 Nov 2006) $
50   */
51  public class RealMatrixImpl implements RealMatrix, Serializable {

138     /***
139      * Compute the sum of this and <code>m</code>.
140      *
141      * @param m    matrix to be added
142      * @return     this + m
143      * @throws  IllegalArgumentException if m is not the same size as this
144      */
145     public RealMatrix add(RealMatrix m) throws IllegalArgumentException {
146         if (this.getColumnDimension() != m.getColumnDimension() ||
147                 this.getRowDimension() != m.getRowDimension()) {
148             throw new IllegalArgumentException("matrix dimension mismatch");
149         }
150         int rowCount = this.getRowDimension();
151         int columnCount = this.getColumnDimension();
152         double[][] outData = new double[rowCount][columnCount];
153         for (int row = 0; row < rowCount; row++) {
154             for (int col = 0; col < columnCount; col++) {
155                 outData[row][col] = data[row][col] + m.getEntry(row, col);
156             }  
157         }
158         return new RealMatrixImpl(outData);
159     }

161     /***
162      * Compute  this minus <code>m</code>.
163      *
164      * @param m    matrix to be subtracted
165      * @return     this + m
166      * @throws  IllegalArgumentException if m is not the same size as *this
167      */
168     public RealMatrix subtract(RealMatrix m) throws IllegalArgumentException {
169         if (this.getColumnDimension() != m.getColumnDimension() ||
170                 this.getRowDimension() != m.getRowDimension()) {
171             throw new IllegalArgumentException("matrix dimension mismatch");
172         }
173         int rowCount = this.getRowDimension();
174         int columnCount = this.getColumnDimension();
175         double[][] outData = new double[rowCount][columnCount];
176         for (int row = 0; row < rowCount; row++) {
177             for (int col = 0; col < columnCount; col++) {
178                 outData[row][col] = data[row][col] - m.getEntry(row, col);
179             }
180         }
181         return new RealMatrixImpl(outData);
182     }

219     /***
220      * Returns the result of postmultiplying this by <code>m</code>.
221      * @param m    matrix to postmultiply by
222      * @return     this*m
223      * @throws     IllegalArgumentException
224      *             if columnDimension(this) != rowDimension(m)
225      */
226     public RealMatrix multiply(RealMatrix m) throws IllegalArgumentException {
227         if (this.getColumnDimension() != m.getRowDimension()) {
228             throw new IllegalArgumentException("Matrices are not multiplication compatible.");
229         }
230         int nRows = this.getRowDimension();
231         int nCols = m.getColumnDimension();
232         int nSum = this.getColumnDimension();
233         double[][] outData = new double[nRows][nCols];
234         double sum = 0;
235         for (int row = 0; row < nRows; row++) {
236             for (int col = 0; col < nCols; col++) {
237                 sum = 0;
238                 for (int i = 0; i < nSum; i++) {
239                     sum += data[row][i] * m.getEntry(i, col);
240                 }
241                 outData[row][col] = sum;
242             }
243         }
244         return new RealMatrixImpl(outData);
245     }

280     /***
281      *
282      * @return norm
283      */
284     public double getNorm() {
285         double maxColSum = 0;
286         for (int col = 0; col < this.getColumnDimension(); col++) {
287             double sum = 0;
288             for (int row = 0; row < this.getRowDimension(); row++) {
289                 sum += Math.abs(data[row][col]);
290             }
291             maxColSum = Math.max(maxColSum, sum);
292         }
293         return maxColSum;
294     }

296     /***
297      * Gets a submatrix. Rows and columns are indicated
298      * counting from 0 to n-1.
299      *
300      * @param startRow Initial row index
301      * @param endRow Final row index
302      * @param startColumn Initial column index
303      * @param endColumn Final column index
304      * @return The subMatrix containing the data of the
305      *         specified rows and columns
306      * @exception MatrixIndexException if row or column selections are not valid
307      */
308     public RealMatrix getSubMatrix(int startRow, int endRow, int startColumn,
309             int endColumn) throws MatrixIndexException {
310         if (startRow < 0 || startRow > endRow || endRow > data.length ||
311              startColumn < 0 || startColumn > endColumn ||
312              endColumn > data[0].length ) {
313             throw new MatrixIndexException(
314                     "invalid row or column index selection");
315         }
316         RealMatrixImpl subMatrix = new RealMatrixImpl(endRow - startRow+1,
317                 endColumn - startColumn+1);
318         double[][] subMatrixData = subMatrix.getDataRef();
319         for (int i = startRow; i <= endRow; i++) {
320             for (int j = startColumn; j <= endColumn; j++) {
321                     subMatrixData[i - startRow][j - startColumn] = data[i][j];
322                 }
323             }
324         return subMatrix;
325     }

327     /***
328      * Gets a submatrix. Rows and columns are indicated
329      * counting from 0 to n-1.
330      *
331      * @param selectedRows Array of row indices must be non-empty
332      * @param selectedColumns Array of column indices must be non-empty
333      * @return The subMatrix containing the data in the
334      *     specified rows and columns
335      * @exception MatrixIndexException  if supplied row or column index arrays
336      *     are not valid
337      */
338     public RealMatrix getSubMatrix(int[] selectedRows, int[] selectedColumns)
339     throws MatrixIndexException {
340         if (selectedRows.length * selectedColumns.length == 0) {
341             throw new MatrixIndexException(
342                     "selected row and column index arrays must be non-empty");
343         }
344         RealMatrixImpl subMatrix = new RealMatrixImpl(selectedRows.length,
345                 selectedColumns.length);
346         double[][] subMatrixData = subMatrix.getDataRef();
347         try  {
348             for (int i = 0; i < selectedRows.length; i++) {
349                 for (int j = 0; j < selectedColumns.length; j++) {
350                     subMatrixData[i][j] = data[selectedRows[i]][selectedColumns[j]];
351                 }
352             }
353         }
354         catch (ArrayIndexOutOfBoundsException e) {
355             throw new MatrixIndexException("matrix dimension mismatch");
356         }
357         return subMatrix;
358     }

360     /***
361      * Replace the submatrix starting at <code>row, column</code> using data in
362      * the input <code>subMatrix</code> array. Indexes are 0-based.
363      * <p> 
364      * Example:<br>
365      * Starting with <pre>
366      * 1  2  3  4
367      * 5  6  7  8
368      * 9  0  1  2
369      * </pre>
370      * and <code>subMatrix = {{3, 4} {5,6}}</code>, invoking 
371      * <code>setSubMatrix(subMatrix,1,1))</code> will result in <pre>
372      * 1  2  3  4
373      * 5  3  4  8
374      * 9  5  6  2
375      * </pre>
376      * 
377      * @param subMatrix  array containing the submatrix replacement data
378      * @param row  row coordinate of the top, left element to be replaced
379      * @param column  column coordinate of the top, left element to be replaced
380      * @throws MatrixIndexException  if subMatrix does not fit into this 
381      *    matrix from element in (row, column) 
382      * @throws IllegalArgumentException if <code>subMatrix</code> is not rectangular
383      *  (not all rows have the same length) or empty
384      * @throws NullPointerException if <code>subMatrix</code> is null
385      * @since 1.1
386      */
387     public void setSubMatrix(double[][] subMatrix, int row, int column) 
388         throws MatrixIndexException {
389         if ((row < 0) || (column < 0)){
390             throw new MatrixIndexException
391                 ("invalid row or column index selection");          
392         }
393         int nRows = subMatrix.length;
394         if (nRows == 0) {
395             throw new IllegalArgumentException(
396             "Matrix must have at least one row."); 
397         }
398         int nCols = subMatrix[0].length;
399         if (nCols == 0) {
400             throw new IllegalArgumentException(
401             "Matrix must have at least one column."); 
402         }
403         for (int r = 1; r < nRows; r++) {
404             if (subMatrix[r].length != nCols) {
405                 throw new IllegalArgumentException(
406                 "All input rows must have the same length.");
407             }
408         }       
409         if (data == null) {
410             if ((row > 0)||(column > 0)) throw new MatrixIndexException
411                 ("matrix must be initialized to perfom this method");
412             data = new double[nRows][nCols];
413             System.arraycopy(subMatrix, 0, data, 0, subMatrix.length);          
414         }   
415         if (((nRows + row) > this.getRowDimension()) ||
416             (nCols + column > this.getColumnDimension()))
417             throw new MatrixIndexException(
418                     "invalid row or column index selection");                   
419         for (int i = 0; i < nRows; i++) {
420             System.arraycopy(subMatrix[i], 0, data[row + i], column, nCols);
421         } 
422         lu = null;
423     }

557     /***
558      * @return determinant
559      * @throws InvalidMatrixException if matrix is not square
560      */
561     public double getDeterminant() throws InvalidMatrixException {
562         if (!isSquare()) {
563             throw new InvalidMatrixException("matrix is not square");
564         }
565         if (isSingular()) {   // note: this has side effect of attempting LU decomp if lu == null
566             return 0d;
567         } else {
568             double det = parity;
569             for (int i = 0; i < this.getRowDimension(); i++) {
570                 det *= lu[i][i];
571             }
572             return det;
573         }
574     }

583     /***
584      * @return true if the matrix is singular
585      */
586     public boolean isSingular() {
587         if (lu == null) {
588             try {
589                 luDecompose();
590                 return false;
591             } catch (InvalidMatrixException ex) {
592                 return true;
593             }
594         } else { // LU decomp must have been successfully performed
595             return false; // so the matrix is not singular
596         }
597     }

628     /***
629      * @param v vector to operate on
630      * @throws IllegalArgumentException if columnDimension != v.length
631      * @return resulting vector
632      */
633     public double[] operate(double[] v) throws IllegalArgumentException {
634         if (v.length != this.getColumnDimension()) {
635             throw new IllegalArgumentException("vector has wrong length");
636         }
637         int nRows = this.getRowDimension();
638         int nCols = this.getColumnDimension();
639         double[] out = new double[v.length];
640         for (int row = 0; row < nRows; row++) {
641             double sum = 0;
642             for (int i = 0; i < nCols; i++) {
643                 sum += data[row][i] * v[i];
644             }
645             out[row] = sum;
646         }
647         return out;
648     }

651      * @param v vector to premultiply by
652      * @throws IllegalArgumentException if rowDimension != v.length
653      * @return resulting matrix
654      */
655     public double[] preMultiply(double[] v) throws IllegalArgumentException {
656         int nRows = this.getRowDimension();
657         if (v.length != nRows) {
658             throw new IllegalArgumentException("vector has wrong length");
659         }
660         int nCols = this.getColumnDimension();
661         double[] out = new double[nCols];
662         for (int col = 0; col < nCols; col++) {
663             double sum = 0;
664             for (int i = 0; i < nRows; i++) {
665                 sum += data[i][col] * v[i];
666             }
667             out[col] = sum;
668         }
669         return out;
670     }

672     /***
673      * Returns a matrix of (column) solution vectors for linear systems with
674      * coefficient matrix = this and constant vectors = columns of
675      * <code>b</code>.
676      *
677      * @param b  array of constant forming RHS of linear systems to
678      * to solve
679      * @return solution array
680      * @throws IllegalArgumentException if this.rowDimension != row dimension
681      * @throws InvalidMatrixException if this matrix is not square or is singular
682      */
683     public double[] solve(double[] b) throws IllegalArgumentException, InvalidMatrixException {
684         int nRows = this.getRowDimension();
685         if (b.length != nRows) {
686             throw new IllegalArgumentException("constant vector has wrong length");
687         }
688         RealMatrix bMatrix = new RealMatrixImpl(b);
689         double[][] solution = ((RealMatrixImpl) (solve(bMatrix))).getDataRef();
690         double[] out = new double[nRows];

697     /***
698      * Returns a matrix of (column) solution vectors for linear systems with
699      * coefficient matrix = this and constant vectors = columns of
700      * <code>b</code>.
701      *
702      * @param b  matrix of constant vectors forming RHS of linear systems to
703      * to solve
704      * @return matrix of solution vectors
705      * @throws IllegalArgumentException if this.rowDimension != row dimension
706      * @throws InvalidMatrixException if this matrix is not square or is singular
707      */
708     public RealMatrix solve(RealMatrix b) throws IllegalArgumentException, InvalidMatrixException  {
709         if (b.getRowDimension() != this.getRowDimension()) {
710             throw new IllegalArgumentException("Incorrect row dimension");
711         }
712         if (!this.isSquare()) {
713             throw new InvalidMatrixException("coefficient matrix is not square");
714         }
715         if (this.isSingular()) { // side effect: compute LU decomp
716             throw new InvalidMatrixException("Matrix is singular.");
717         }
718 
719         int nCol = this.getColumnDimension();
720         int nColB = b.getColumnDimension();
721         int nRowB = b.getRowDimension();
722 
723         // Apply permutations to b
724         double[][] bp = new double[nRowB][nColB];
725         for (int row = 0; row < nRowB; row++) {
726             for (int col = 0; col < nColB; col++) {
727                 bp[row][col] = b.getEntry(permutation[row], col);
728             }
729         }
730 
731         // Solve LY = b
732         for (int col = 0; col < nCol; col++) {
733             for (int i = col + 1; i < nCol; i++) {
734                 for (int j = 0; j < nColB; j++) {
735                     bp[i][j] -= bp[col][j] * lu[i][col];
736                 }
737             }
738         }
739 
740         // Solve UX = Y
741         for (int col = nCol - 1; col >= 0; col--) {
742             for (int j = 0; j < nColB; j++) {
743                 bp[col][j] /= lu[col][col];
744             }
745             for (int i = 0; i < col; i++) {
746                 for (int j = 0; j < nColB; j++) {
747                     bp[i][j] -= bp[col][j] * lu[i][col];
748                 }
749             }
750         }
751 
752         RealMatrixImpl outMat = new RealMatrixImpl(bp);
753         return outMat;
754     }

756     /***
757      * Computes a new
758      * <a href="http://www.math.gatech.edu/~bourbaki/math2601/Web-notes/2num.pdf">
759      * LU decompostion</a> for this matrix, storing the result for use by other methods.
760      * <p>
761      * <strong>Implementation Note</strong>:<br>
762      * Uses <a href="http://www.damtp.cam.ac.uk/user/fdl/people/sd/lectures/nummeth98/linear.htm">
763      * Crout's algortithm</a>, with partial pivoting.
764      * <p>
765      * <strong>Usage Note</strong>:<br>
766      * This method should rarely be invoked directly. Its only use is
767      * to force recomputation of the LU decomposition when changes have been
768      * made to the underlying data using direct array references. Changes
769      * made using setXxx methods will trigger recomputation when needed
770      * automatically.
771      *
772      * @throws InvalidMatrixException if the matrix is non-square or singular.
773      */
774     public void luDecompose() throws InvalidMatrixException {
775 
776         int nRows = this.getRowDimension();
777         int nCols = this.getColumnDimension();
778         if (nRows != nCols) {
779             throw new InvalidMatrixException("LU decomposition requires that the matrix be square.");
780         }
781         lu = this.getData();
782 
783         // Initialize permutation array and parity
784         permutation = new int[nRows];
785         for (int row = 0; row < nRows; row++) {
786             permutation[row] = row;
787         }
788         parity = 1;
789 
790         // Loop over columns
791         for (int col = 0; col < nCols; col++) {
792 
793             double sum = 0;
794 
795             // upper
796             for (int row = 0; row < col; row++) {
797                 sum = lu[row][col];
798                 for (int i = 0; i < row; i++) {
799                     sum -= lu[row][i] * lu[i][col];
800                 }
801                 lu[row][col] = sum;
802             }
803 
804             // lower
805             int max = col; // permutation row
806             double largest = 0d;
807             for (int row = col; row < nRows; row++) {
808                 sum = lu[row][col];
809                 for (int i = 0; i < col; i++) {
810                     sum -= lu[row][i] * lu[i][col];
811                 }
812                 lu[row][col] = sum;
813 
814                 // maintain best permutation choice
815                 if (Math.abs(sum) > largest) {
816                     largest = Math.abs(sum);
817                     max = row;
818                 }
819             }
820 
821             // Singularity check
822             if (Math.abs(lu[max][col]) < TOO_SMALL) {
823                 lu = null;
824                 throw new InvalidMatrixException("matrix is singular");
825             }
826 
827             // Pivot if necessary
828             if (max != col) {
829                 double tmp = 0;
830                 for (int i = 0; i < nCols; i++) {
831                     tmp = lu[max][i];
832                     lu[max][i] = lu[col][i];
833                     lu[col][i] = tmp;
834                 }
835                 int temp = permutation[max];
836                 permutation[max] = permutation[col];
837                 permutation[col] = temp;
838                 parity = -parity;
839             }
840 
841             //Divide the lower elements by the "winning" diagonal elt.
842             for (int row = col + 1; row < nRows; row++) {
843                 lu[row][col] /= lu[col][col];
844             }
845         }
846     }
