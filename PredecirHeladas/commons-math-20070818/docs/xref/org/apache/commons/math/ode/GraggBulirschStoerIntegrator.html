20  /***
21   * This class implements a Gragg-Bulirsch-Stoer integrator for
22   * Ordinary Differential Equations.
23  
24   * <p>The Gragg-Bulirsch-Stoer algorithm is one of the most efficient
25   * ones currently available for smooth problems. It uses Richardson
26   * extrapolation to estimate what would be the solution if the step
27   * size could be decreased down to zero.</p>
28  
29   * <p>
30   * This method changes both the step size and the order during
31   * integration, in order to minimize computation cost. It is
32   * particularly well suited when a very high precision is needed. The
33   * limit where this method becomes more efficient than high-order
34   * Runge-Kutta-Fehlberg methods like {@link DormandPrince853Integrator
35   * Dormand-Prince 8(5,3)} depends on the problem. Results given in the
36   * Hairer, Norsett and Wanner book show for example that this limit
37   * occurs for accuracy around 1e-6 when integrating Saltzam-Lorenz
38   * equations (the authors note this problem is <i>extremely sensitive
39   * to the errors in the first integration steps</i>), and around 1e-11
40   * for a two dimensional celestial mechanics problems with seven
41   * bodies (pleiades problem, involving quasi-collisions for which
42   * <i>automatic step size control is essential</i>).
43   * </p>
44  
45   * <p>
46   * This implementation is basically a reimplementation in Java of the
47   * <a
48   * href="http://www.unige.ch/math/folks/hairer/prog/nonstiff/odex.f">odex</a>
49   * fortran code by E. Hairer and G. Wanner. The redistribution policy
50   * for this code is available <a
51   * href="http://www.unige.ch/~hairer/prog/licence.txt">here</a>, for
52   * convenience, it is reproduced below.</p>
53   * </p>

138   /*** Set the stability check controls.
139    * <p>The stability check is performed on the first few iterations of
140    * the extrapolation scheme. If this test fails, the step is rejected
141    * and the stepsize is reduced.</p>
142    * <p>By default, the test is performed, at most during two
143    * iterations at each step, and at most once for each of these
144    * iterations. The default stepsize reduction factor is 0.5.</p>
145    * @param performTest if true, stability check will be performed,
146      if false, the check will be skipped
147    * @param maxIter maximal number of iterations for which checks are
148    * performed (the number of iterations is reset to default if negative
149    * or null)
150    * @param maxChecks maximal number of checks for each iteration
151    * (the number of checks is reset to default if negative or null)
152    * @param stabilityReduction stepsize reduction factor in case of
153    * failure (the factor is reset to default if lower than 0.0001 or
154    * greater than 0.9999)
155    */
156   public void setStabilityCheck(boolean performTest,
157                                 int maxIter, int maxChecks,
158                                 double stabilityReduction) {
159 
160     this.performTest = performTest;
161     this.maxIter     = (maxIter   <= 0) ? 2 : maxIter;
162     this.maxChecks   = (maxChecks <= 0) ? 1 : maxChecks;
163 
164     if ((stabilityReduction < 0.0001) || (stabilityReduction > 0.9999)) {
165       this.stabilityReduction = 0.5;
166     } else {
167       this.stabilityReduction = stabilityReduction;
168     }
169 
170   }

172   /*** Set the step size control factors.
173 
174    * <p>The new step size hNew is computed from the old one h by:
175    * <pre>
176    * hNew = h * stepControl2 / (err/stepControl1)^(1/(2k+1))
177    * </pre>
178    * where err is the scaled error and k the iteration number of the
179    * extrapolation scheme (counting from 0). The default values are
180    * 0.65 for stepControl1 and 0.94 for stepControl2.</p>
181    * <p>The step size is subject to the restriction:
182    * <pre>
183    * stepControl3^(1/(2k+1))/stepControl4 <= hNew/h <= 1/stepControl3^(1/(2k+1))
184    * </pre>
185    * The default values are 0.02 for stepControl3 and 4.0 for
186    * stepControl4.</p>
187    * @param stepControl1 first stepsize control factor (the factor is
188    * reset to default if lower than 0.0001 or greater than 0.9999)
189    * @param stepControl2 second stepsize control factor (the factor
190    * is reset to default if lower than 0.0001 or greater than 0.9999)
191    * @param stepControl3 third stepsize control factor (the factor is
192    * reset to default if lower than 0.0001 or greater than 0.9999)
193    * @param stepControl4 fourth stepsize control factor (the factor
194    * is reset to default if lower than 1.0001 or greater than 999.9)
195    */
196   public void setStepsizeControl(double stepControl1, double stepControl2,
197                                  double stepControl3, double stepControl4) {
198 
199     if ((stepControl1 < 0.0001) || (stepControl1 > 0.9999)) {
200       this.stepControl1 = 0.65;
201     } else {
202       this.stepControl1 = stepControl1;
203     }
204 
205     if ((stepControl2 < 0.0001) || (stepControl2 > 0.9999)) {
206       this.stepControl2 = 0.94;
207     } else {
208       this.stepControl2 = stepControl2;
209     }
210 
211     if ((stepControl3 < 0.0001) || (stepControl3 > 0.9999)) {
212       this.stepControl3 = 0.02;
213     } else {
214       this.stepControl3 = stepControl3;
215     }
216 
217     if ((stepControl4 < 1.0001) || (stepControl4 > 999.9)) {
218       this.stepControl4 = 4.0;
219     } else {
220       this.stepControl4 = stepControl4;
221     }
222 
223   }

225   /*** Set the order control parameters.
226    * <p>The Gragg-Bulirsch-Stoer method changes both the step size and
227    * the order during integration, in order to minimize computation
228    * cost. Each extrapolation step increases the order by 2, so the
229    * maximal order that will be used is always even, it is twice the
230    * maximal number of columns in the extrapolation table.</p>
231    * <pre>
232    * order is decreased if w(k-1) <= w(k)   * orderControl1
233    * order is increased if w(k)   <= w(k-1) * orderControl2
234    * </pre>
235    * <p>where w is the table of work per unit step for each order
236    * (number of function calls divided by the step length), and k is
237    * the current order.</p>
238    * <p>The default maximal order after construction is 18 (i.e. the
239    * maximal number of columns is 9). The default values are 0.8 for
240    * orderControl1 and 0.9 for orderControl2.</p>
241    * @param maxOrder maximal order in the extrapolation table (the
242    * maximal order is reset to default if order <= 6 or odd)
243    * @param orderControl1 first order control factor (the factor is
244    * reset to default if lower than 0.0001 or greater than 0.9999)
245    * @param orderControl2 second order control factor (the factor
246    * is reset to default if lower than 0.0001 or greater than 0.9999)
247    */
248   public void setOrderControl(int maxOrder,
249                               double orderControl1, double orderControl2) {
250 
251     if ((maxOrder <= 6) || (maxOrder % 2 != 0)) {
252       this.maxOrder = 18;
253     }
254 
255     if ((orderControl1 < 0.0001) || (orderControl1 > 0.9999)) {
256       this.orderControl1 = 0.8;
257     } else {
258       this.orderControl1 = orderControl1;
259     }
260 
261     if ((orderControl2 < 0.0001) || (orderControl2 > 0.9999)) {
262       this.orderControl2 = 0.9;
263     } else {
264       this.orderControl2 = orderControl2;
265     }
266 
267     // reinitialize the arrays
268     initializeArrays();
269 
270   }

307   /*** Initialize the integrator internal arrays. */
308   private void initializeArrays() {
309 
310     int size = maxOrder / 2;
311 
312     if ((sequence == null) || (sequence.length != size)) {
313       // all arrays should be reallocated with the right size
314       sequence        = new int[size];
315       costPerStep     = new int[size];
316       coeff           = new double[size][];
317       costPerTimeUnit = new double[size];
318       optimalStep     = new double[size];
319     }
320 
321     if (denseOutput) {
322       // step size sequence: 2, 6, 10, 14, ...
323       for (int k = 0; k < size; ++k) {
324         sequence[k] = 4 * k + 2;
325       }
326     } else {
327       // step size sequence: 2, 4, 6, 8, ...
328       for (int k = 0; k < size; ++k) {
329         sequence[k] = 2 * (k + 1); 
330       }
331     }
332 
333     // initialize the order selection cost array
334     // (number of function calls for each column of the extrapolation table)
335     costPerStep[0] = sequence[0] + 1;
336     for (int k = 1; k < size; ++k) {
337       costPerStep[k] = costPerStep[k-1] + sequence[k];
338     }
339 
340     // initialize the extrapolation tables
341     for (int k = 0; k < size; ++k) {
342       coeff[k] = (k > 0) ? new double[k] : null;
343       for (int l = 0; l < k; ++l) {
344         double ratio = ((double) sequence[k]) / sequence[k-l-1];
345         coeff[k][l] = 1.0 / (ratio * ratio - 1.0);
346       }
347     }
348 
349   }

381   /*** Update scaling array.
382    * @param y1 first state vector to use for scaling
383    * @param y2 second state vector to use for scaling
384    * @param scale scaling array to update
385    */
386   private void rescale(double[] y1, double[] y2, double[] scale) {
387     if (vecAbsoluteTolerance == null) {
388       for (int i = 0; i < scale.length; ++i) {
389         double yi = Math.max(Math.abs(y1[i]), Math.abs(y2[i]));
390         scale[i] = scalAbsoluteTolerance + scalRelativeTolerance * yi;
391       }
392     } else {
393       for (int i = 0; i < scale.length; ++i) {
394         double yi = Math.max(Math.abs(y1[i]), Math.abs(y2[i]));
395         scale[i] = vecAbsoluteTolerance[i] + vecRelativeTolerance[i] * yi;
396       }
397     }
398   }

400   /*** Perform integration over one step using substeps of a modified
401    * midpoint method.
402    * @param equations differential equations to integrate
403    * @param t0 initial time
404    * @param y0 initial value of the state vector at t0
405    * @param step global step
406    * @param k iteration number (from 0 to sequence.length - 1)
407    * @param scale scaling array
408    * @param f placeholder where to put the state vector derivatives at each substep
409    *          (element 0 already contains initial derivative)
410    * @param yMiddle placeholder where to put the state vector at the middle of the step
411    * @param yEnd placeholder where to put the state vector at the end
412    * @param yTmp placeholder for one state vector
413    * @return true if computation was done properly,
414    *         false if stability check failed before end of computation
415    * @throws DerivativeException this exception is propagated to the caller if the
416    * underlying user function triggers one
417    */
418   private boolean tryStep(FirstOrderDifferentialEquations equations,
419                           double t0, double[] y0, double step, int k,
420                           double[] scale, double[][] f,
421                           double[] yMiddle, double[] yEnd,
422                           double[] yTmp)
423     throws DerivativeException {
424 
425     int    n        = sequence[k];
426     double subStep  = step / n;
427     double subStep2 = 2 * subStep;
428 
429     // first substep
430     double t = t0 + subStep;
431     for (int i = 0; i < y0.length; ++i) {
432       yTmp[i] = y0[i];
433       yEnd[i] = y0[i] + subStep * f[0][i];
434     }
435     equations.computeDerivatives(t, yEnd, f[1]);
436 
437     // other substeps
438     for (int j = 1; j < n; ++j) {
439 
440       if (2 * j == n) {
441         // save the point at the middle of the step
442         System.arraycopy(yEnd, 0, yMiddle, 0, y0.length);
443       }
444 
445       t += subStep;
446       for (int i = 0; i < y0.length; ++i) {
447         double middle = yEnd[i];
448         yEnd[i]       = yTmp[i] + subStep2 * f[j][i];
449         yTmp[i]       = middle;
450       }
451 
452       equations.computeDerivatives(t, yEnd, f[j+1]);
453 
454       // stability check
455       if (performTest && (j <= maxChecks) && (k < maxIter)) {
456         double initialNorm = 0.0;
457         for (int l = 0; l < y0.length; ++l) {
458           double ratio = f[0][l] / scale[l];
459           initialNorm += ratio * ratio;
460         }
461         double deltaNorm = 0.0;
462         for (int l = 0; l < y0.length; ++l) {
463           double ratio = (f[j+1][l] - f[0][l]) / scale[l];
464           deltaNorm += ratio * ratio;
465         }
466         if (deltaNorm > 4 * Math.max(1.0e-15, initialNorm)) {
467           return false;
468         }
469       }
470 
471     }
472 
473     // correction of the last substep (at t0 + step)
474     for (int i = 0; i < y0.length; ++i) {
475       yEnd[i] = 0.5 * (yTmp[i] + yEnd[i] + subStep * f[n][i]);
476     }
477 
478     return true;
479 
480   }

482   /*** Extrapolate a vector.
483    * @param offset offset to use in the coefficients table
484    * @param k index of the last updated point
485    * @param diag working diagonal of the Aitken-Neville's
486    * triangle, without the last element
487    * @param last last element
488    */
489   private void extrapolate(int offset, int k, double[][] diag, double[] last) {
490 
491     // update the diagonal
492     for (int j = 1; j < k; ++j) {
493       for (int i = 0; i < last.length; ++i) {
494         // Aitken-Neville's recursive formula
495         diag[k-j-1][i] = diag[k-j][i]
496                        + coeff[k+offset][j-1] * (diag[k-j][i] - diag[k-j-1][i]);
497       }
498     }
499 
500     // update the last element
501     for (int i = 0; i < last.length; ++i) {
502       // Aitken-Neville's recursive formula
503       last[i] = diag[0][i] + coeff[k+offset][k-1] * (diag[0][i] - last[i]);
504     }
505   }

507   public void integrate(FirstOrderDifferentialEquations equations,
508                         double t0, double[] y0, double t, double[] y)
509   throws DerivativeException, IntegratorException {
510 
511     // sanity check
512     if (equations.getDimension() != y0.length) {
513       throw new IntegratorException("dimensions mismatch: "
514                                     + "ODE problem has dimension {0}"
515                                     + ", state vector has dimension {1}",
516                                     new String[] {
517                                       Integer.toString(equations.getDimension()),
518                                       Integer.toString(y0.length)
519                                     });
520     }
521     if (Math.abs(t - t0) <= 1.0e-12 * Math.max(Math.abs(t0), Math.abs(t))) {
522       throw new IntegratorException("too small integration interval: length = {0}",
523                                     new String[] {
524                                       Double.toString(Math.abs(t - t0))
525                                     });
526     }
527 
528     boolean forward = (t > t0);
529 
530     // create some internal working arrays
531     double[] yDot0   = new double[y0.length];
532     double[] y1      = new double[y0.length];
533     double[] yTmp    = new double[y0.length];
534     double[] yTmpDot = new double[y0.length];
535 
536     double[][] diagonal = new double[sequence.length-1][];
537     double[][] y1Diag = new double[sequence.length-1][];
538     for (int k = 0; k < sequence.length-1; ++k) {
539       diagonal[k] = new double[y0.length];
540       y1Diag[k] = new double[y0.length];
541     }
542 
543     double[][][] fk  = new double[sequence.length][][];
544     for (int k = 0; k < sequence.length; ++k) {
545 
546       fk[k]    = new double[sequence[k] + 1][];
547 
548       // all substeps start at the same point, so share the first array
549       fk[k][0] = yDot0;
550 
551       for (int l = 0; l < sequence[k]; ++l) {
552         fk[k][l+1] = new double[y0.length];
553       }
554 
555     }
556 
557     if (y != y0) {
558       System.arraycopy(y0, 0, y, 0, y0.length);
559     }
560 
561     double[] yDot1      = null;
562     double[][] yMidDots = null;
563     if (denseOutput) {
564       yDot1    = new double[y0.length];
565       yMidDots = new double[1 + 2 * sequence.length][];
566       for (int j = 0; j < yMidDots.length; ++j) {
567         yMidDots[j] = new double[y0.length];
568       }
569     } else {
570       yMidDots    = new double[1][];
571       yMidDots[0] = new double[y0.length];
572     }
573 
574     // initial scaling
575     double[] scale = new double[y0.length];
576     rescale(y, y, scale);
577 
578     // initial order selection
579     double log10R = Math.log(Math.max(1.0e-10,
580                                       (vecRelativeTolerance == null)
581                                       ? scalRelativeTolerance
582                                       : vecRelativeTolerance[0]))
583                   / Math.log(10.0);
584     int targetIter = Math.max(1,
585                               Math.min(sequence.length - 2,
586                                        (int) Math.floor(0.5 - 0.6 * log10R)));
587     // set up an interpolator sharing the integrator arrays
588     AbstractStepInterpolator interpolator = null;
589     if (denseOutput || (! switchesHandler.isEmpty())) {
590       interpolator = new GraggBulirschStoerStepInterpolator(y, yDot0,
591                                                             y1, yDot1,
592                                                             yMidDots, forward);
593     } else {
594       interpolator = new DummyStepInterpolator(y, forward);
595     }
596     interpolator.storeTime(t0);
597 
598     double  currentT         = t0;
599     double  hNew             = 0;
600     double  maxError         = Double.MAX_VALUE;
601     boolean previousRejected = false;
602     boolean firstTime        = true;
603     boolean newStep          = true;
604     boolean lastStep         = false;
605     boolean firstStepAlreadyComputed = false;
606     handler.reset();
607     costPerTimeUnit[0] = 0;
608     while (! lastStep) {
609 
610       double h;
611       double error;
612       boolean reject = false;
613 
614       if (newStep) {
615 
616         interpolator.shift();
617 
618         // first evaluation, at the beginning of the step
619         if (! firstStepAlreadyComputed) {
620           equations.computeDerivatives(currentT, y, yDot0);
621         }
622 
623         if (firstTime) {
624 
625           hNew = initializeStep(equations, forward,
626                                 2 * targetIter + 1, scale,
627                                 currentT, y, yDot0, yTmp, yTmpDot);
628 
629           if (! forward) {
630             hNew = -hNew;
631           }
632 
633         }
634 
635         newStep = false;
636 
637       }
638 
639       h = hNew;
640 
641       // step adjustment near bounds
642       if ((forward && (currentT + h > t))
643           || ((! forward) && (currentT + h < t))) {
644         h = t - currentT;
645       }
646       double nextT = currentT + h;
647       lastStep = forward ? (nextT >= t) : (nextT <= t);
648 
649       // iterate over several substep sizes
650       int k = -1;
651       for (boolean loop = true; loop; ) {
652 
653         ++k;
654 
655         // modified midpoint integration with the current substep
656         if ( ! tryStep(equations, currentT, y, h, k, scale, fk[k],
657                        (k == 0) ? yMidDots[0] : diagonal[k-1],
658                        (k == 0) ? y1 : y1Diag[k-1],
659                        yTmp)) {
660 
661           // the stability check failed, we reduce the global step
662           hNew   = Math.abs(filterStep(h * stabilityReduction, false));
663           reject = true;
664           loop   = false;
665 
666         } else {
667 
668           // the substep was computed successfully
669           if (k > 0) {
670 
671             // extrapolate the state at the end of the step
672             // using last iteration data
673             extrapolate(0, k, y1Diag, y1);
674             rescale(y, y1, scale);
675 
676             // estimate the error at the end of the step.
677             error = 0;
678             for (int j = 0; j < y0.length; ++j) {
679               double e = Math.abs(y1[j] - y1Diag[0][j]) / scale[j];
680               error += e * e;
681             }
682             error = Math.sqrt(error / y0.length);
683 
684             if ((error > 1.0e15) || ((k > 1) && (error > maxError))) {
685               // error is too big, we reduce the global step
686               hNew   = Math.abs(filterStep(h * stabilityReduction, false));
687               reject = true;
688               loop   = false;
689             } else {
690 
691               maxError = Math.max(4 * error, 1.0);
692 
693               // compute optimal stepsize for this order
694               double exp = 1.0 / (2 * k + 1);
695               double fac = stepControl2 / Math.pow(error / stepControl1, exp);
696               double pow = Math.pow(stepControl3, exp);
697               fac = Math.max(pow / stepControl4, Math.min(1 / pow, fac));
698               optimalStep[k]     = Math.abs(filterStep(h * fac, true));
699               costPerTimeUnit[k] = costPerStep[k] / optimalStep[k];
700 
701               // check convergence
702               switch (k - targetIter) {
703 
704               case -1 :
705                 if ((targetIter > 1) && ! previousRejected) {
706 
707                   // check if we can stop iterations now
708                   if (error <= 1.0) {
709                     // convergence have been reached just before targetIter
710                     loop = false;
711                   } else {
712                     // estimate if there is a chance convergence will
713                     // be reached on next iteration, using the
714                     // asymptotic evolution of error
715                     double ratio = ((double) sequence [k] * sequence[k+1])
716                                  / (sequence[0] * sequence[0]);
717                     if (error > ratio * ratio) {
718                       // we don't expect to converge on next iteration
719                       // we reject the step immediately and reduce order
720                       reject = true;
721                       loop   = false;
722                       targetIter = k;
723                       if ((targetIter > 1)
724                           && (costPerTimeUnit[targetIter-1]
725                               < orderControl1 * costPerTimeUnit[targetIter])) {
726                         --targetIter;
727                       }
728                       hNew = optimalStep[targetIter];
729                     }
730                   }
731                 }
732                 break;
733 
734               case 0:
735                 if (error <= 1.0) {
736                   // convergence has been reached exactly at targetIter
737                   loop = false;
738                 } else {
739                   // estimate if there is a chance convergence will
740                   // be reached on next iteration, using the
741                   // asymptotic evolution of error
742                   double ratio = ((double) sequence[k+1]) / sequence[0];
743                   if (error > ratio * ratio) {
744                     // we don't expect to converge on next iteration
745                     // we reject the step immediately
746                     reject = true;
747                     loop = false;
748                     if ((targetIter > 1)
749                         && (costPerTimeUnit[targetIter-1]
750                             < orderControl1 * costPerTimeUnit[targetIter])) {
751                       --targetIter;
752                     }
753                     hNew = optimalStep[targetIter];
754                   }
755                 }
756                 break;
757 
758               case 1 :
759                 if (error > 1.0) {
760                   reject = true;
761                   if ((targetIter > 1)
762                       && (costPerTimeUnit[targetIter-1]
763                           < orderControl1 * costPerTimeUnit[targetIter])) {
764                     --targetIter;
765                   }
766                   hNew = optimalStep[targetIter];
767                 }
768                 loop = false;
769                 break;
770 
771               default :
772                 if ((firstTime || lastStep) && (error <= 1.0)) {
773                   loop = false;
774                 }
775                 break;
776 
777               }
778 
779             }
780           }
781         }
782       }
783 
784       // dense output handling
785       double hInt = getMaxStep();
786       if (denseOutput && ! reject) {
787 
788         // extrapolate state at middle point of the step
789         for (int j = 1; j <= k; ++j) {
790           extrapolate(0, j, diagonal, yMidDots[0]);
791         }
792 
793         // derivative at end of step
794         equations.computeDerivatives(currentT + h, y1, yDot1);
795 
796         int mu = 2 * k - mudif + 3;
797 
798         for (int l = 0; l < mu; ++l) {
799 
800           // derivative at middle point of the step
801           int l2 = l / 2;
802           double factor = Math.pow(0.5 * sequence[l2], l);
803           int middleIndex = fk[l2].length / 2;
804           for (int i = 0; i < y0.length; ++i) {
805             yMidDots[l+1][i] = factor * fk[l2][middleIndex + l][i];
806           }
807           for (int j = 1; j <= k - l2; ++j) {
808             factor = Math.pow(0.5 * sequence[j + l2], l);
809             middleIndex = fk[l2+j].length / 2;
810             for (int i = 0; i < y0.length; ++i) {
811               diagonal[j-1][i] = factor * fk[l2+j][middleIndex+l][i];
812             }
813             extrapolate(l2, j, diagonal, yMidDots[l+1]);
814           }
815           for (int i = 0; i < y0.length; ++i) {
816             yMidDots[l+1][i] *= h;
817           }
818 
819           // compute centered differences to evaluate next derivatives
820           for (int j = (l + 1) / 2; j <= k; ++j) {
821             for (int m = fk[j].length - 1; m >= 2 * (l + 1); --m) {
822               for (int i = 0; i < y0.length; ++i) {
823                 fk[j][m][i] -= fk[j][m-2][i];
824               }
825             }
826           }
827 
828         }
829 
830         if (mu >= 0) {
831 
832           // estimate the dense output coefficients
833           GraggBulirschStoerStepInterpolator gbsInterpolator
834             = (GraggBulirschStoerStepInterpolator) interpolator;
835           gbsInterpolator.computeCoefficients(mu, h);
836 
837           if (useInterpolationError) {
838             // use the interpolation error to limit stepsize
839             double interpError = gbsInterpolator.estimateError(scale);
840             hInt = Math.abs(h / Math.max(Math.pow(interpError, 1.0 / (mu+4)),
841                                          0.01));
842             if (interpError > 10.0) {
843               hNew = hInt;
844               reject = true;
845             }
846           }
847 
848           // Switching functions handling
849           if (!reject) {
850             interpolator.storeTime(currentT + h);
851             if (switchesHandler.evaluateStep(interpolator)) {
852               reject = true;
853               hNew = Math.abs(switchesHandler.getEventTime() - currentT);
854             }
855           }
856 
857         }
858 
859         if (!reject) {
860           // we will reuse the slope for the beginning of next step
861           firstStepAlreadyComputed = true;
862           System.arraycopy(yDot1, 0, yDot0, 0, y0.length);
863         }
864 
865       }
866 
867       if (! reject) {
868 
869         // store end of step state
870         currentT += h;
871         System.arraycopy(y1, 0, y, 0, y0.length);
872 
873         switchesHandler.stepAccepted(currentT, y);
874         if (switchesHandler.stop()) {
875           lastStep = true;
876         }
877 
878         // provide the step data to the step handler
879         interpolator.storeTime(currentT);
880         handler.handleStep(interpolator, lastStep);
881 
882         if (switchesHandler.reset(currentT, y) && ! lastStep) {
883           // some switching function has triggered changes that
884           // invalidate the derivatives, we need to recompute them
885           firstStepAlreadyComputed = false;
886         }
887 
888         int optimalIter;
889         if (k == 1) {
890           optimalIter = 2;
891           if (previousRejected) {
892             optimalIter = 1;
893           }
894         } else if (k <= targetIter) {
895           optimalIter = k;
896           if (costPerTimeUnit[k-1] < orderControl1 * costPerTimeUnit[k]) {
897             optimalIter = k-1;
898           } else if (costPerTimeUnit[k] < orderControl2 * costPerTimeUnit[k-1]) {
899             optimalIter = Math.min(k+1, sequence.length - 2);
900           }
901         } else {
902           optimalIter = k - 1;
903           if ((k > 2)
904               && (costPerTimeUnit[k-2] < orderControl1 * costPerTimeUnit[k-1])) {
905             optimalIter = k - 2;
906           }
907           if (costPerTimeUnit[k] < orderControl2 * costPerTimeUnit[optimalIter]) {
908             optimalIter = Math.min(k, sequence.length - 2);
909           }
910         }
911 
912         if (previousRejected) {
913           // after a rejected step neither order nor stepsize
914           // should increase
915           targetIter = Math.min(optimalIter, k);
916           hNew = Math.min(Math.abs(h), optimalStep[targetIter]);
917         } else {
918           // stepsize control
919           if (optimalIter <= k) {
920             hNew = optimalStep[optimalIter];
921           } else {
922             if ((k < targetIter)
923                 && (costPerTimeUnit[k] < orderControl2 * costPerTimeUnit[k-1])) {
924               hNew = filterStep(optimalStep[k]
925                                 * costPerStep[optimalIter+1] / costPerStep[k],
926                                 false);
927             } else {
928               hNew = filterStep(optimalStep[k]
929                                 * costPerStep[optimalIter] / costPerStep[k],
930                                 false);
931             }
932           }
933 
934           targetIter = optimalIter;
935 
936         }
937 
938         newStep = true;
939 
940       }
941 
942       hNew = Math.min(hNew, hInt);
943       if (! forward) {
944         hNew = -hNew;
945       }
946 
947       firstTime = false;
948 
949       if (reject) {
950         lastStep = false;
951         previousRejected = true;
952       } else {
953         previousRejected = false;
954       }
