55       * 
56       * @param x an addend
57       * @param y an addend

59       * @throws ArithmeticException if the result can not be represented as an

64          long s = (long)x + (long)y;

71      /***
72       * Returns an exact representation of the <a
73       * href="http://mathworld.wolfram.com/BinomialCoefficient.html"> Binomial
74       * Coefficient</a>, "<code>n choose k</code>", the number of
75       * <code>k</code>-element subsets that can be selected from an
76       * <code>n</code>-element set.
77       * <p>
78       * <Strong>Preconditions</strong>:
79       * <ul>
80       * <li> <code>0 <= k <= n </code> (otherwise
81       * <code>IllegalArgumentException</code> is thrown)</li>
82       * <li> The result is small enough to fit into a <code>long</code>. The
83       * largest value of <code>n</code> for which all coefficients are
84       * <code> < Long.MAX_VALUE</code> is 66. If the computed value exceeds
85       * <code>Long.MAX_VALUE</code> an <code>ArithMeticException
86       *      </code> is
87       * thrown.</li>
88       * </ul>
89       * 
90       * @param n the size of the set
91       * @param k the size of the subsets to be counted
92       * @return <code>n choose k</code>
93       * @throws IllegalArgumentException if preconditions are not met.
94       * @throws ArithmeticException if the result is too large to be represented
95       *         by a long integer.
96       */
97      public static long binomialCoefficient(final int n, final int k) {
98          if (n < k) {
99              throw new IllegalArgumentException(
100                 "must have n >= k for binomial coefficient (n,k)");
101         }
102         if (n < 0) {
103             throw new IllegalArgumentException(
104                 "must have n >= 0 for binomial coefficient (n,k)");
105         }
106         if ((n == k) || (k == 0)) {
107             return 1;
108         }
109         if ((k == 1) || (k == n - 1)) {
110             return n;
111         }
112 
113         long result = Math.round(binomialCoefficientDouble(n, k));
114         if (result == Long.MAX_VALUE) {
115             throw new ArithmeticException(
116                 "result too large to represent in a long integer");
117         }
118         return result;
119     }

166     /***
167      * Returns the natural <code>log</code> of the <a
168      * href="http://mathworld.wolfram.com/BinomialCoefficient.html"> Binomial
169      * Coefficient</a>, "<code>n choose k</code>", the number of
170      * <code>k</code>-element subsets that can be selected from an
171      * <code>n</code>-element set.
172      * <p>
173      * <Strong>Preconditions</strong>:
174      * <ul>
175      * <li> <code>0 <= k <= n </code> (otherwise
176      * <code>IllegalArgumentException</code> is thrown)</li>
177      * </ul>
178      * 
179      * @param n the size of the set
180      * @param k the size of the subsets to be counted
181      * @return <code>n choose k</code>
182      * @throws IllegalArgumentException if preconditions are not met.
183      */
184     public static double binomialCoefficientLog(final int n, final int k) {
185         if (n < k) {
186             throw new IllegalArgumentException(
187                 "must have n >= k for binomial coefficient (n,k)");
188         }
189         if (n < 0) {
190             throw new IllegalArgumentException(
191                 "must have n >= 0 for binomial coefficient (n,k)");
192         }
193         if ((n == k) || (k == 0)) {
194             return 0;
195         }
196         if ((k == 1) || (k == n - 1)) {
197             return Math.log((double)n);
198         }
199         double logSum = 0;
200 
201         // n!/k!
202         for (int i = k + 1; i <= n; i++) {
203             logSum += Math.log((double)i);
204         }
205 
206         // divide by (n-k)!
207         for (int i = 2; i <= n - k; i++) {
208             logSum -= Math.log((double)i);
209         }
210 
211         return logSum;
212     }

273      * <Strong>Preconditions</strong>:
274      * <ul>
275      * <li> <code>n >= 0</code> (otherwise
276      * <code>IllegalArgumentException</code> is thrown)</li>
277      * <li> The result is small enough to fit into a <code>double</code>. The

295     /***
296      * Returns the natural logarithm of n!.
297      * <p>
298      * <Strong>Preconditions</strong>:
299      * <ul>
300      * <li> <code>n >= 0</code> (otherwise
301      * <code>IllegalArgumentException</code> is thrown)</li>

303      * 
304      * @param n argument
305      * @return <code>n!</code>
306      * @throws IllegalArgumentException if preconditions are not met.

308     public static double factorialLog(final int n) {
309         if (n < 0) {
310             throw new IllegalArgumentException("must have n > 0 for n!");
311         }
312         double logSum = 0;
313         for (int i = 2; i <= n; i++) {
314             logSum += Math.log((double)i);
315         }
316         return logSum;
317     }

319     /***
320      * <p>
321      * Gets the greatest common divisor of the absolute value of two numbers,
322      * using the "binary gcd" method which avoids division and modulo
323      * operations. See Knuth 4.5.2 algorithm B. This algorithm is due to Josef
324      * Stein (1961).
325      * </p>
326      * 
327      * @param u a non-zero number
328      * @param v a non-zero number
329      * @return the greatest common divisor, never zero
330      * @since 1.1
331      */
332     public static int gcd(int u, int v) {
333         if (u * v == 0) {
334             return (Math.abs(u) + Math.abs(v));
335         }
336         // keep u and v negative, as negative integers range down to
337         // -2^31, while positive numbers can only be as large as 2^31-1
338         // (i.e. we can't necessarily negate a negative number without
339         // overflow)
340         /* assert u!=0 && v!=0; */
341         if (u > 0) {
342             u = -u;
343         } // make u negative
344         if (v > 0) {
345             v = -v;
346         } // make v negative
347         // B1. [Find power of 2]
348         int k = 0;
349         while ((u & 1) == 0 && (v & 1) == 0 && k < 31) { // while u and v are
350                                                             // both even...
351             u /= 2;
352             v /= 2;
353             k++; // cast out twos.
354         }
355         if (k == 31) {
356             throw new ArithmeticException("overflow: gcd is 2^31");
357         }
358         // B2. Initialize: u and v have been divided by 2^k and at least
359         // one is odd.
360         int t = ((u & 1) == 1) ? v : -(u / 2)/* B3 */;
361         // t negative: u was odd, v may be even (t replaces v)
362         // t positive: u was even, v is odd (t replaces u)
363         do {
364             /* assert u<0 && v<0; */
365             // B4/B3: cast out twos from t.
366             while ((t & 1) == 0) { // while t is even..
367                 t /= 2; // cast out twos
368             }
369             // B5 [reset max(u,v)]
370             if (t > 0) {
371                 u = -t;
372             } else {
373                 v = t;
374             }
375             // B6/B3. at this point both u and v should be odd.
376             t = (v - u) / 2;
377             // |u| larger: t positive (replace u)
378             // |v| larger: t negative (replace v)
379         } while (t != 0);
380         return -u * (1 << k); // gcd is u*2^k
381     }

387      * @return the hash code

389     public static int hash(double value) {

496     /***
497      * Get the next machine representable number after a number, moving
498      * in the direction of another number.
499      * <p>
500      * If <code>direction</code> is greater than or equal to<code>d</code>,
501      * the smallest machine representable number strictly greater than
502      * <code>d</code> is returned; otherwise the largest representable number
503      * strictly less than <code>d</code> is returned.
504      * <p>
505      * If <code>d</code> is NaN or Infinite, it is returned unchanged.
506      * 
507      * @param d base number
508      * @param direction (the only important thing is whether
509      * direction is greater or smaller than d)
510      * @return the next machine representable number in the specified direction
511      */
512     public static double nextAfter(double d, double direction) {
513 
514         // handling of some important special cases
515         if (Double.isNaN(d) || Double.isInfinite(d)) {
516                 return d;
517         } else if (d == 0) {
518                 return (direction < 0) ? -Double.MIN_VALUE : Double.MIN_VALUE;
519         }
520         // special cases MAX_VALUE to infinity and  MIN_VALUE to 0
521         // are handled just as normal numbers
522 
523         // split the double in raw components
524         long bits     = Double.doubleToLongBits(d);
525         long sign     = bits & 0x8000000000000000L;
526         long exponent = bits & 0x7ff0000000000000L;
527         long mantissa = bits & 0x000fffffffffffffL;
528 
529         if (d * (direction - d) >= 0) {
530                 // we should increase the mantissa
531                 if (mantissa == 0x000fffffffffffffL) {
532                         return Double.longBitsToDouble(sign |
533                                         (exponent + 0x0010000000000000L));
534                 } else {
535                         return Double.longBitsToDouble(sign |
536                                         exponent | (mantissa + 1));
537                 }
538         } else {
539                 // we should decrease the mantissa
540                 if (mantissa == 0L) {
541                         return Double.longBitsToDouble(sign |
542                                         (exponent - 0x0010000000000000L) |
543                                         0x000fffffffffffffL);
544                 } else {
545                         return Double.longBitsToDouble(sign |
546                                         exponent | (mantissa - 1));
547                 }
548         }
549 
550     }

623     /***
624      * Round the given non-negative, value to the "nearest" integer. Nearest is
625      * determined by the rounding method specified. Rounding methods are defined
626      * in {@link BigDecimal}.
627      * 
628      * @param unscaled the value to round.
629      * @param sign the sign of the original, scaled value.
630      * @param roundingMethod the rounding method as defined in
631      *        {@link BigDecimal}.
632      * @return the rounded value.
633      * @since 1.1
634      */
635     private static double roundUnscaled(double unscaled, double sign,
636         int roundingMethod) {
637         switch (roundingMethod) {
638         case BigDecimal.ROUND_CEILING :
639             if (sign == -1) {
640                 unscaled = Math.floor(nextAfter(unscaled, Double.NEGATIVE_INFINITY));
641             } else {
642                 unscaled = Math.ceil(nextAfter(unscaled, Double.POSITIVE_INFINITY));
643             }
644             break;
645         case BigDecimal.ROUND_DOWN :
646             unscaled = Math.floor(nextAfter(unscaled, Double.NEGATIVE_INFINITY));
647             break;
648         case BigDecimal.ROUND_FLOOR :
649             if (sign == -1) {
650                 unscaled = Math.ceil(nextAfter(unscaled, Double.POSITIVE_INFINITY));
651             } else {
652                 unscaled = Math.floor(nextAfter(unscaled, Double.NEGATIVE_INFINITY));
653             }
654             break;
655         case BigDecimal.ROUND_HALF_DOWN : {
656             unscaled = nextAfter(unscaled, Double.NEGATIVE_INFINITY);
657             double fraction = unscaled - Math.floor(unscaled);
658             if (fraction > 0.5) {
659                 unscaled = Math.ceil(unscaled);
660             } else {
661                 unscaled = Math.floor(unscaled);
662             }
663             break;
664         }
665         case BigDecimal.ROUND_HALF_EVEN : {
666             double fraction = unscaled - Math.floor(unscaled);
667             if (fraction > 0.5) {
668                 unscaled = Math.ceil(unscaled);
669             } else if (fraction < 0.5) {
670                 unscaled = Math.floor(unscaled);
671             } else {
672                 if (Math.floor(unscaled) / 2.0 == Math.floor(Math
673                     .floor(unscaled) / 2.0)) { // even
674                     unscaled = Math.floor(unscaled);
675                 } else { // odd
676                     unscaled = Math.ceil(unscaled);
677                 }
678             }
679             break;
680         }
681         case BigDecimal.ROUND_HALF_UP : {
682             unscaled = nextAfter(unscaled, Double.POSITIVE_INFINITY);
683             double fraction = unscaled - Math.floor(unscaled);
684             if (fraction >= 0.5) {
685                 unscaled = Math.ceil(unscaled);
686             } else {
687                 unscaled = Math.floor(unscaled);
688             }
689             break;
690         }
691         case BigDecimal.ROUND_UNNECESSARY :
692             if (unscaled != Math.floor(unscaled)) {
693                 throw new ArithmeticException("Inexact result from rounding");
694             }
695             break;
696         case BigDecimal.ROUND_UP :
697             unscaled = Math.ceil(nextAfter(unscaled,  Double.POSITIVE_INFINITY));
698             break;
699         default :
700             throw new IllegalArgumentException("Invalid rounding method.");
701         }
702         return unscaled;
703     }
