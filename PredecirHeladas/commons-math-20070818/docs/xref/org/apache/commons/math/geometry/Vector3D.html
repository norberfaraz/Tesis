75    /*** Simple constructor.
76     * Build a vector from its azimuthal coordinates
77     * @param alpha azimuth (&alpha;) around Z
78     *              (0 is +X, &pi;/2 is +Y, &pi; is -X and 3&pi;/2 is -Y)
79     * @param delta elevation (&delta;) above (XY) plane, from -&pi;/2 to +&pi;/2
80     * @see #getAlpha()
81     * @see #getDelta()
82     */
83    public Vector3D(double alpha, double delta) {
84      double cosDelta = Math.cos(delta);
85      this.x = Math.cos(alpha) * cosDelta;
86      this.y = Math.sin(alpha) * cosDelta;
87      this.z = Math.sin(delta);
88    }

233   /*** Normalize the instance.
234    * @return a new normalized vector
235    * @exception ArithmeticException if the norm is null
236    */
237   public Vector3D normalize() {
238     double s = getNorm();
239     if (s == 0) {
240       throw new ArithmeticException("null norm");
241     }
242     return multiply(1 / s);
243   }

245   /*** Get a vector orthogonal to the instance.
246    * <p>There are an infinite number of normalized vectors orthogonal
247    * to the instance. This method picks up one of them almost
248    * arbitrarily. It is useful when one needs to compute a reference
249    * frame with one of the axes in a predefined direction. The
250    * following example shows how to build a frame having the k axis
251    * aligned with the known vector u :
252    * <pre><code>
253    *   Vector3D k = u.normalize();
254    *   Vector3D i = k.orthogonal();
255    *   Vector3D j = Vector3D.crossProduct(k, i);
256    * </code></pre></p>
257    * @return a new normalized vector orthogonal to the instance
258    * @exception ArithmeticException if the norm of the instance is null
259    */
260   public Vector3D orthogonal() {
261 
262     double threshold = 0.6 * getNorm();
263     if (threshold == 0) {
264       throw new ArithmeticException("null norm");
265     }
266 
267     if ((x >= -threshold) && (x <= threshold)) {
268       double inverse  = 1 / Math.sqrt(y * y + z * z);
269       return new Vector3D(0, inverse * z, -inverse * y);
270     } else if ((y >= -threshold) && (y <= threshold)) {
271       double inverse  = 1 / Math.sqrt(x * x + z * z);
272       return new Vector3D(-inverse * z, 0, inverse * x);
273     } else {
274       double inverse  = 1 / Math.sqrt(x * x + y * y);
275       return new Vector3D(inverse * y, -inverse * x, 0);
276     }
277 
278   }

280   /*** Compute the angular separation between two vectors.
281    * <p>This method computes the angular separation between two
282    * vectors using the dot product for well separated vectors and the
283    * cross product for almost aligned vectors. This allow to have a
284    * good accuracy in all cases, even for vectors very close to each
285    * other.</p>
286    * @param v1 first vector
287    * @param v2 second vector
288    * @exception ArithmeticException if either vector has a null norm
289    */
290   public static double angle(Vector3D v1, Vector3D v2) {
291 
292     double normProduct = v1.getNorm() * v2.getNorm();
293     if (normProduct == 0) {
294       throw new ArithmeticException("null norm");
295     }
296 
297     double dot = dotProduct(v1, v2);
298     double threshold = normProduct * 0.9999;
299     if ((dot < -threshold) || (dot > threshold)) {
300       // the vectors are almost aligned, compute using the sine
301       Vector3D v3 = crossProduct(v1, v2);
302       if (dot >= 0) {
303         return Math.asin(v3.getNorm() / normProduct);
304       }
305       return Math.PI - Math.asin(v3.getNorm() / normProduct);
306     }
307     
308     // the vectors are sufficiently separated to use the cosine
309     return Math.acos(dot / normProduct);
310 
311   }

328   /*** Compute the dot-product of two vectors.
329    * @param v1 first vector
330    * @param v2 second vector
331    * @return the dot product v1.v2
332    */
333   public static double dotProduct(Vector3D v1, Vector3D v2) {
334     return v1.x * v2.x + v1.y * v2.y + v1.z * v2.z;
335   }

337   /*** Compute the cross-product of two vectors.
338    * @param v1 first vector
339    * @param v2 second vector
340    * @return the cross product v1 ^ v2 as a new Vector
341    */
342   public static Vector3D crossProduct(Vector3D v1, Vector3D v2) {
343     return new Vector3D(v1.y * v2.z - v1.z * v2.y,
344                         v1.z * v2.x - v1.x * v2.z,
345                         v1.x * v2.y - v1.y * v2.x);
346   }
