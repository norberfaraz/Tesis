22  /*** 
23   * This class solves a least squares problem.
24   *
25   * <p>This implementation <em>should</em> work even for over-determined systems
26   * (i.e. systems having more variables than equations). Over-determined systems
27   * are solved by ignoring the variables which have the smallest impact according
28   * to their jacobian column norm. Only the rank of the matrix and some loop bounds
29   * are changed to implement this. This feature has undergone only basic testing
30   * for now and should still be considered experimental.</p>

33   * href="http://www.netlib.org/minpack/lmder.f">lmder</a> routine with minor

35   * decomposition which has been rewritten following the algorithm described in the

191   /*** 
192    * Update the jacobian matrix.
193    */
194   private void updateJacobian() {
195     ++jacobianEvaluations;
196     Arrays.fill(jacobian, 0);
197     for (int i = 0, index = 0; i < rows; i++) {
198       WeightedMeasurement wm = measurements[i];
199       double factor = -Math.sqrt(wm.getWeight());
200       for (int j = 0; j < cols; ++j) {
201         jacobian[index++] = factor * wm.getPartial(parameters[j]);
202       }
203     }
204   }
205 
206   /*** 
207    * Update the residuals array and cost function value.
208    */
209   private void updateResidualsAndCost() {
210     ++costEvaluations;
211     cost = 0;
212     for (int i = 0, index = 0; i < rows; i++, index += cols) {
213       WeightedMeasurement wm = measurements[i];
214       double residual = wm.getResidual();
215       residuals[i] = Math.sqrt(wm.getWeight()) * residual;
216       cost += wm.getWeight() * residual * residual;
217     }
218     cost = Math.sqrt(cost);
219   }
220 
221   /*** 
222    * Get the Root Mean Square value.
223    * Get the Root Mean Square value, i.e. the root of the arithmetic
224    * mean of the square of all weighted residuals. This is related to the
225    * criterion that is minimized by the estimator as follows: if
226    * <em>c</em> if the criterion, and <em>n</em> is the number of
227    * measurements, then the RMS is <em>sqrt (c/n)</em>.
228    * 
229    * @param problem estimation problem
230    * @return RMS value
231    */
232   public double getRMS(EstimationProblem problem) {
233     WeightedMeasurement[] wm = problem.getMeasurements();
234     double criterion = 0;
235     for (int i = 0; i < wm.length; ++i) {
236       double residual = wm[i].getResidual();
237       criterion += wm[i].getWeight() * residual * residual;
238     }
239     return Math.sqrt(criterion / wm.length);
240   }

242   /*** 
243    * Solve an estimation problem using the Levenberg-Marquardt algorithm.
244    * <p>The algorithm used is a modified Levenberg-Marquardt one, based
245    * on the MINPACK <a href="http://www.netlib.org/minpack/lmder.f">lmder</a>
246    * routine. The algorithm settings must have been set up before this method
247    * is called with the {@link #setInitialStepBoundFactor},
248    * {@link #setMaxCostEval}, {@link #setCostRelativeTolerance},
249    * {@link #setParRelativeTolerance} and {@link #setOrthoTolerance} methods.
250    * If these methods have not been called, the default values set up by the
251    * {@link #LevenbergMarquardtEstimator() constructor} will be used.</p>
252    * <p>The authors of the original fortran function are:</p>
253    * <ul>
254    *   <li>Argonne National Laboratory. MINPACK project. March 1980</li>
255    *   <li>Burton  S. Garbow</li>
256    *   <li>Kenneth E. Hillstrom</li>
257    *   <li>Jorge   J. More</li>
258    *   </ul>
259    * <p>Luc Maisonobe did the Java translation.</p>
260    * 
261    * @param problem estimation problem to solve
262    * @exception EstimationException if convergence cannot be
263    * reached with the specified algorithm settings or if there are more variables
264    * than equations
265    * @see #setInitialStepBoundFactor
266    * @see #setMaxCostEval
267    * @see #setCostRelativeTolerance
268    * @see #setParRelativeTolerance
269    * @see #setOrthoTolerance
270    */
271   public void estimate(EstimationProblem problem)
272     throws EstimationException {
273 
274     // retrieve the equations and the parameters
275     measurements = problem.getMeasurements();
276     parameters   = problem.getUnboundParameters();
277 
278     // arrays shared with the other private methods
279     rows        = measurements.length;
280     cols        = parameters.length;
281     solvedCols  = Math.min(rows, cols);
282     jacobian    = new double[rows * cols];
283     diagR       = new double[cols];
284     jacNorm     = new double[cols];
285     beta        = new double[cols];
286     permutation = new int[cols];
287     lmDir       = new double[cols];
288     residuals   = new double[rows];
289 
290     // local variables
291     double   delta   = 0, xNorm = 0;
292     double[] diag    = new double[cols];
293     double[] oldX    = new double[cols];
294     double[] oldRes  = new double[rows];
295     double[] work1   = new double[cols];
296     double[] work2   = new double[cols];
297     double[] work3   = new double[cols];
298 
299     // evaluate the function at the starting point and calculate its norm
300     updateResidualsAndCost();
301     
302     // outer loop
303     lmPar               = 0;
304     costEvaluations     = 0;
305     jacobianEvaluations = 0;
306     boolean firstIteration = true;
307     while (costEvaluations < maxCostEval) {
308 
309       // compute the Q.R. decomposition of the jacobian matrix
310       updateJacobian();
311       qrDecomposition();
312 
313       // compute Qt.res
314       qTy(residuals);
315 
316       // now we don't need Q anymore,
317       // so let jacobian contain the R matrix with its diagonal elements
318       for (int k = 0; k < solvedCols; ++k) {
319         int pk = permutation[k];
320         jacobian[k * cols + pk] = diagR[pk];
321       }
322 
323       if (firstIteration) {
324 
325         // scale the variables according to the norms of the columns
326         // of the initial jacobian
327         xNorm = 0;
328         for (int k = 0; k < cols; ++k) {
329           double dk = jacNorm[k];
330           if (dk == 0) {
331             dk = 1.0;
332           }
333           double xk = dk * parameters[k].getEstimate();
334           xNorm  += xk * xk;
335           diag[k] = dk;
336         }
337         xNorm = Math.sqrt(xNorm);
338         
339         // initialize the step bound delta
340         delta = (xNorm == 0)
341               ? initialStepBoundFactor : (initialStepBoundFactor * xNorm);
342  
343       }
344 
345       // check orthogonality between function vector and jacobian columns
346       double maxCosine = 0;
347       if (cost != 0) {
348         for (int j = 0; j < solvedCols; ++j) {
349           int    pj = permutation[j];
350           double s  = jacNorm[pj];
351           if (s != 0) {
352             double sum = 0;
353             for (int i = 0, index = pj; i <= j; ++i, index += cols) {
354               sum += jacobian[index] * residuals[i];
355             }
356             maxCosine = Math.max(maxCosine, Math.abs(sum) / (s * cost));
357           }
358         }
359       }
360       if (maxCosine <= orthoTolerance) {
361         return;
362       }
363 
364       // rescale if necessary
365       for (int j = 0; j < cols; ++j) {
366         diag[j] = Math.max(diag[j], jacNorm[j]);
367       }
368 
369       // inner loop
370       for (double ratio = 0; ratio < 1.0e-4;) {
371 
372         // save the state
373         for (int j = 0; j < solvedCols; ++j) {
374           int pj = permutation[j];
375           oldX[pj] = parameters[pj].getEstimate();
376         }
377         double previousCost = cost;
378         double[] tmpVec = residuals;
379         residuals = oldRes;
380         oldRes    = tmpVec;
381         
382         // determine the Levenberg-Marquardt parameter
383         determineLMParameter(oldRes, delta, diag, work1, work2, work3);
384 
385         // compute the new point and the norm of the evolution direction
386         double lmNorm = 0;
387         for (int j = 0; j < solvedCols; ++j) {
388           int pj = permutation[j];
389           lmDir[pj] = -lmDir[pj];
390           parameters[pj].setEstimate(oldX[pj] + lmDir[pj]);
391           double s = diag[pj] * lmDir[pj];
392           lmNorm  += s * s;
393         }
394         lmNorm = Math.sqrt(lmNorm);
395 
396         // on the first iteration, adjust the initial step bound.
397         if (firstIteration) {
398           delta = Math.min(delta, lmNorm);
399         }
400 
401         // evaluate the function at x + p and calculate its norm
402         updateResidualsAndCost();
403 
404         // compute the scaled actual reduction
405         double actRed = -1.0;
406         if (0.1 * cost < previousCost) {
407           double r = cost / previousCost;
408           actRed = 1.0 - r * r;
409         }
410 
411         // compute the scaled predicted reduction
412         // and the scaled directional derivative
413         for (int j = 0; j < solvedCols; ++j) {
414           int pj = permutation[j];
415           double dirJ = lmDir[pj];
416           work1[j] = 0;
417           for (int i = 0, index = pj; i <= j; ++i, index += cols) {
418             work1[i] += jacobian[index] * dirJ;
419           }
420         }
421         double coeff1 = 0;
422         for (int j = 0; j < solvedCols; ++j) {
423          coeff1 += work1[j] * work1[j];
424         }
425         double pc2 = previousCost * previousCost;
426         coeff1 = coeff1 / pc2;
427         double coeff2 = lmPar * lmNorm * lmNorm / pc2;
428         double preRed = coeff1 + 2 * coeff2;
429         double dirDer = -(coeff1 + coeff2);
430 
431         // ratio of the actual to the predicted reduction
432         ratio = (preRed == 0) ? 0 : (actRed / preRed);
433 
434         // update the step bound
435         if (ratio <= 0.25) {
436           double tmp =
437             (actRed < 0) ? (0.5 * dirDer / (dirDer + 0.5 * actRed)) : 0.5;
438           if ((0.1 * cost >= previousCost) || (tmp < 0.1)) {
439             tmp = 0.1;
440           }
441           delta = tmp * Math.min(delta, 10.0 * lmNorm);
442           lmPar /= tmp;
443         } else if ((lmPar == 0) || (ratio >= 0.75)) {
444           delta = 2 * lmNorm;
445           lmPar *= 0.5;
446         }
447 
448         // test for successful iteration.
449         if (ratio >= 1.0e-4) {
450           // successful iteration, update the norm
451           firstIteration = false;
452           xNorm = 0;
453           for (int k = 0; k < cols; ++k) {
454             double xK = diag[k] * parameters[k].getEstimate();
455             xNorm    += xK * xK;
456           }
457           xNorm = Math.sqrt(xNorm);
458         } else {
459           // failed iteration, reset the previous values
460           cost = previousCost;
461           for (int j = 0; j < solvedCols; ++j) {
462             int pj = permutation[j];
463             parameters[pj].setEstimate(oldX[pj]);
464           }
465           tmpVec    = residuals;
466           residuals = oldRes;
467           oldRes    = tmpVec;
468         }
469    
470         // tests for convergence.
471         if (((Math.abs(actRed) <= costRelativeTolerance)
472             && (preRed <= costRelativeTolerance)
473             && (ratio <= 2.0))
474             || (delta <= parRelativeTolerance * xNorm)) {
475           return;
476         }
477 
478         // tests for termination and stringent tolerances
479         // (2.2204e-16 is the machine epsilon for IEEE754)
480         if (costEvaluations >= maxCostEval) {
481           break;
482         }
483         if ((Math.abs(actRed) <= 2.2204e-16)
484             && (preRed <= 2.2204e-16)
485             && (ratio <= 2.0)) {
486           throw new EstimationException("cost relative tolerance is too small ({0}),"
487                                       + " no further reduction in the"
488                                       + " sum of squares is possible",
489                                         new String[] {
490                                           Double.toString(costRelativeTolerance)
491                                         });
492         } else if (delta <= 2.2204e-16 * xNorm) {
493           throw new EstimationException("parameters relative tolerance is too small"
494                                       + " ({0}), no further improvement in"
495                                       + " the approximate solution is possible",
496                                         new String[] {
497                                           Double.toString(parRelativeTolerance)
498                                         });
499         } else if (maxCosine <= 2.2204e-16)  {
500           throw new EstimationException("orthogonality tolerance is too small ({0}),"
501                                       + " solution is orthogonal to the jacobian",
502                                         new String[] {
503                                           Double.toString(orthoTolerance)
504                                         });
505         }
506 
507       }
508 
509     }
510 
511     throw new EstimationException("maximal number of evaluations exceeded ({0})",
512                                   new String[] {
513                                     Integer.toString(maxCostEval)
514                                   });
515 
516   }

518   /*** 
519    * Determine the Levenberg-Marquardt parameter.
520    * <p>This implementation is a translation in Java of the MINPACK
521    * <a href="http://www.netlib.org/minpack/lmpar.f">lmpar</a>
522    * routine.</p>
523    * <p>This method sets the lmPar and lmDir attributes.</p>
524    * <p>The authors of the original fortran function are:</p>
525    * <ul>
526    *   <li>Argonne National Laboratory. MINPACK project. March 1980</li>
527    *   <li>Burton  S. Garbow</li>
528    *   <li>Kenneth E. Hillstrom</li>
529    *   <li>Jorge   J. More</li>
530    * </ul>
531    * <p>Luc Maisonobe did the Java translation.</p>
532    * 
533    * @param qy array containing qTy
534    * @param delta upper bound on the euclidean norm of diagR * lmDir
535    * @param diag diagonal matrix
536    * @param work1 work array
537    * @param work2 work array
538    * @param work3 work array
539    */
540   private void determineLMParameter(double[] qy, double delta, double[] diag,
541                                     double[] work1, double[] work2, double[] work3) {
542 
543     // compute and store in x the gauss-newton direction, if the
544     // jacobian is rank-deficient, obtain a least squares solution
545     for (int j = 0; j < rank; ++j) {
546       lmDir[permutation[j]] = qy[j];
547     }
548     for (int j = rank; j < cols; ++j) {
549       lmDir[permutation[j]] = 0;
550     }
551     for (int k = rank - 1; k >= 0; --k) {
552       int pk = permutation[k];
553       double ypk = lmDir[pk] / diagR[pk];
554       for (int i = 0, index = pk; i < k; ++i, index += cols) {
555         lmDir[permutation[i]] -= ypk * jacobian[index];
556       }
557       lmDir[pk] = ypk;
558     }
559 
560     // evaluate the function at the origin, and test
561     // for acceptance of the Gauss-Newton direction
562     double dxNorm = 0;
563     for (int j = 0; j < solvedCols; ++j) {
564       int pj = permutation[j];
565       double s = diag[pj] * lmDir[pj];
566       work1[pj] = s;
567       dxNorm += s * s;
568     }
569     dxNorm = Math.sqrt(dxNorm);
570     double fp = dxNorm - delta;
571     if (fp <= 0.1 * delta) {
572       lmPar = 0;
573       return;
574     }
575 
576     // if the jacobian is not rank deficient, the Newton step provides
577     // a lower bound, parl, for the zero of the function,
578     // otherwise set this bound to zero
579     double sum2, parl = 0;
580     if (rank == solvedCols) {
581       for (int j = 0; j < solvedCols; ++j) {
582         int pj = permutation[j];
583         work1[pj] *= diag[pj] / dxNorm; 
584       }
585       sum2 = 0;
586       for (int j = 0; j < solvedCols; ++j) {
587         int pj = permutation[j];
588         double sum = 0;
589         for (int i = 0, index = pj; i < j; ++i, index += cols) {
590           sum += jacobian[index] * work1[permutation[i]];
591         }
592         double s = (work1[pj] - sum) / diagR[pj];
593         work1[pj] = s;
594         sum2 += s * s;
595       }
596       parl = fp / (delta * sum2);
597     }
598 
599     // calculate an upper bound, paru, for the zero of the function
600     sum2 = 0;
601     for (int j = 0; j < solvedCols; ++j) {
602       int pj = permutation[j];
603       double sum = 0;
604       for (int i = 0, index = pj; i <= j; ++i, index += cols) {
605         sum += jacobian[index] * qy[i];
606       }
607       sum /= diag[pj];
608       sum2 += sum * sum;
609     }
610     double gNorm = Math.sqrt(sum2);
611     double paru = gNorm / delta;
612     if (paru == 0) {
613       // 2.2251e-308 is the smallest positive real for IEE754
614       paru = 2.2251e-308 / Math.min(delta, 0.1);
615     }
616 
617     // if the input par lies outside of the interval (parl,paru),
618     // set par to the closer endpoint
619     lmPar = Math.min(paru, Math.max(lmPar, parl));
620     if (lmPar == 0) {
621       lmPar = gNorm / dxNorm;
622     }
623 
624     for (int countdown = 10; countdown >= 0; --countdown) {
625 
626       // evaluate the function at the current value of lmPar
627       if (lmPar == 0) {
628         lmPar = Math.max(2.2251e-308, 0.001 * paru);
629       }
630       double sPar = Math.sqrt(lmPar);
631       for (int j = 0; j < solvedCols; ++j) {
632         int pj = permutation[j];
633         work1[pj] = sPar * diag[pj];
634       }
635       determineLMDirection(qy, work1, work2, work3);
636 
637       dxNorm = 0;
638       for (int j = 0; j < solvedCols; ++j) {
639         int pj = permutation[j];
640         double s = diag[pj] * lmDir[pj];
641         work3[pj] = s;
642         dxNorm += s * s;
643       }
644       dxNorm = Math.sqrt(dxNorm);
645       double previousFP = fp;
646       fp = dxNorm - delta;
647 
648       // if the function is small enough, accept the current value
649       // of lmPar, also test for the exceptional cases where parl is zero
650       if ((Math.abs(fp) <= 0.1 * delta)
651           || ((parl == 0) && (fp <= previousFP) && (previousFP < 0))) {
652         return;
653       }
654  
655       // compute the Newton correction
656       for (int j = 0; j < solvedCols; ++j) {
657        int pj = permutation[j];
658         work1[pj] = work3[pj] * diag[pj] / dxNorm; 
659       }
660       for (int j = 0; j < solvedCols; ++j) {
661         int pj = permutation[j];
662         work1[pj] /= work2[j];
663         double tmp = work1[pj];
664         for (int i = j + 1; i < solvedCols; ++i) {
665           work1[permutation[i]] -= jacobian[i * cols + pj] * tmp;
666         }
667       }
668       sum2 = 0;
669       for (int j = 0; j < solvedCols; ++j) {
670         double s = work1[permutation[j]];
671         sum2 += s * s;
672       }
673       double correction = fp / (delta * sum2);
674 
675       // depending on the sign of the function, update parl or paru.
676       if (fp > 0) {
677         parl = Math.max(parl, lmPar);
678       } else if (fp < 0) {
679         paru = Math.min(paru, lmPar);
680       }
681 
682       // compute an improved estimate for lmPar
683       lmPar = Math.max(parl, lmPar + correction);
684 
685     }
686   }

688   /*** 
689    * Solve a*x = b and d*x = 0 in the least squares sense.
690    * <p>This implementation is a translation in Java of the MINPACK
691    * <a href="http://www.netlib.org/minpack/qrsolv.f">qrsolv</a>
692    * routine.</p>
693    * <p>This method sets the lmDir and lmDiag attributes.</p>
694    * <p>The authors of the original fortran function are:</p>
695    * <ul>
696    *   <li>Argonne National Laboratory. MINPACK project. March 1980</li>
697    *   <li>Burton  S. Garbow</li>
698    *   <li>Kenneth E. Hillstrom</li>
699    *   <li>Jorge   J. More</li>
700    * </ul>
701    * <p>Luc Maisonobe did the Java translation.</p>
702    * 
703    * @param qy array containing qTy
704    * @param diag diagonal matrix
705    * @param lmDiag diagonal elements associated with lmDir
706    * @param work work array
707    */
708   private void determineLMDirection(double[] qy, double[] diag,
709                                     double[] lmDiag, double[] work) {
710 
711     // copy R and Qty to preserve input and initialize s
712     //  in particular, save the diagonal elements of R in lmDir
713     for (int j = 0; j < solvedCols; ++j) {
714       int pj = permutation[j];
715       for (int i = j + 1; i < solvedCols; ++i) {
716         jacobian[i * cols + pj] = jacobian[j * cols + permutation[i]];
717       }
718       lmDir[j] = diagR[pj];
719       work[j]  = qy[j];
720     }
721 
722     // eliminate the diagonal matrix d using a Givens rotation
723     for (int j = 0; j < solvedCols; ++j) {
724 
725       // prepare the row of d to be eliminated, locating the
726       // diagonal element using p from the Q.R. factorization
727       int pj = permutation[j];
728       double dpj = diag[pj];
729       if (dpj != 0) {
730         Arrays.fill(lmDiag, j + 1, lmDiag.length, 0);
731       }
732       lmDiag[j] = dpj;
733 
734       //  the transformations to eliminate the row of d
735       // modify only a single element of Qty
736       // beyond the first n, which is initially zero.
737       double qtbpj = 0;
738       for (int k = j; k < solvedCols; ++k) {
739         int pk = permutation[k];
740 
741         // determine a Givens rotation which eliminates the
742         // appropriate element in the current row of d
743         if (lmDiag[k] != 0) {
744 
745           double sin, cos;
746           double rkk = jacobian[k * cols + pk];
747           if (Math.abs(rkk) < Math.abs(lmDiag[k])) {
748             double cotan = rkk / lmDiag[k];
749             sin   = 1.0 / Math.sqrt(1.0 + cotan * cotan);
750             cos   = sin * cotan;
751           } else {
752             double tan = lmDiag[k] / rkk;
753             cos = 1.0 / Math.sqrt(1.0 + tan * tan);
754             sin = cos * tan;
755           }
756 
757           // compute the modified diagonal element of R and
758           // the modified element of (Qty,0)
759           jacobian[k * cols + pk] = cos * rkk + sin * lmDiag[k];
760           double temp = cos * work[k] + sin * qtbpj;
761           qtbpj = -sin * work[k] + cos * qtbpj;
762           work[k] = temp;
763 
764           // accumulate the tranformation in the row of s
765           for (int i = k + 1; i < solvedCols; ++i) {
766             double rik = jacobian[i * cols + pk];
767             temp = cos * rik + sin * lmDiag[i];
768             lmDiag[i] = -sin * rik + cos * lmDiag[i];
769             jacobian[i * cols + pk] = temp;
770           }
771 
772         }
773       }
774 
775       // store the diagonal element of s and restore
776       // the corresponding diagonal element of R
777       int index = j * cols + permutation[j];
778       lmDiag[j]       = jacobian[index];
779       jacobian[index] = lmDir[j];
780 
781     }
782 
783     // solve the triangular system for z, if the system is
784     // singular, then obtain a least squares solution
785     int nSing = solvedCols;
786     for (int j = 0; j < solvedCols; ++j) {
787       if ((lmDiag[j] == 0) && (nSing == solvedCols)) {
788         nSing = j;
789       }
790       if (nSing < solvedCols) {
791         work[j] = 0;
792       }
793     }
794     if (nSing > 0) {
795       for (int j = nSing - 1; j >= 0; --j) {
796         int pj = permutation[j];
797         double sum = 0;
798         for (int i = j + 1; i < nSing; ++i) {
799           sum += jacobian[i * cols + pj] * work[i];
800         }
801         work[j] = (work[j] - sum) / lmDiag[j];
802       }
803     }
804 
805     // permute the components of z back to components of lmDir
806     for (int j = 0; j < lmDir.length; ++j) {
807       lmDir[permutation[j]] = work[j];
808     }
809 
810   }

812   /*** 
813    * Decompose a matrix A as A.P = Q.R using Householder transforms.
814    * <p>As suggested in the P. Lascaux and R. Theodor book
815    * <i>Analyse num&eacute;rique matricielle appliqu&eacute;e &agrave;
816    * l'art de l'ing&eacute;nieur</i> (Masson, 1986), instead of representing
817    * the Householder transforms with u<sub>k</sub> unit vectors such that:
818    * <pre>
819    * H<sub>k</sub> = I - 2u<sub>k</sub>.u<sub>k</sub><sup>t</sup>
820    * </pre>
821    * we use <sub>k</sub> non-unit vectors such that:
822    * <pre>
823    * H<sub>k</sub> = I - beta<sub>k</sub>v<sub>k</sub>.v<sub>k</sub><sup>t</sup>
824    * </pre>
825    * where v<sub>k</sub> = a<sub>k</sub> - alpha<sub>k</sub> e<sub>k</sub>.
826    * The beta<sub>k</sub> coefficients are provided upon exit as recomputing
827    * them from the v<sub>k</sub> vectors would be costly.</p>
828    * <p>This decomposition handles rank deficient cases since the tranformations
829    * are performed in non-increasing columns norms order thanks to columns
830    * pivoting. The diagonal elements of the R matrix are therefore also in
831    * non-increasing absolute values order.</p>
832    */
833   private void qrDecomposition() {
834 
835     // initializations
836     for (int k = 0; k < cols; ++k) {
837       permutation[k] = k;
838       double norm2 = 0;
839       for (int index = k; index < jacobian.length; index += cols) {
840         double akk = jacobian[index];
841         norm2 += akk * akk;
842       }
843       jacNorm[k] = Math.sqrt(norm2);
844     }
845 
846     // transform the matrix column after column
847     for (int k = 0; k < cols; ++k) {
848 
849       // select the column with the greatest norm on active components
850       int nextColumn = -1;
851       double ak2 = Double.NEGATIVE_INFINITY;
852       for (int i = k; i < cols; ++i) {
853         double norm2 = 0;
854         int iDiag = k * cols + permutation[i];
855         for (int index = iDiag; index < jacobian.length; index += cols) {
856           double aki = jacobian[index];
857           norm2 += aki * aki;
858         }
859         if (norm2 > ak2) {
860           nextColumn = i;
861           ak2        = norm2;
862         }
863       }
864       if (ak2 == 0) {
865         rank = k;
866         return;
867       }
868       int pk                  = permutation[nextColumn];
869       permutation[nextColumn] = permutation[k];
870       permutation[k]          = pk;
871 
872       // choose alpha such that Hk.u = alpha ek
873       int    kDiag = k * cols + pk;
874       double akk   = jacobian[kDiag];
875       double alpha = (akk > 0) ? -Math.sqrt(ak2) : Math.sqrt(ak2);
876       double betak = 1.0 / (ak2 - akk * alpha);
877       beta[pk]     = betak;
878 
879       // transform the current column
880       diagR[pk]        = alpha;
881       jacobian[kDiag] -= alpha;
882 
883       // transform the remaining columns
884       for (int dk = cols - 1 - k; dk > 0; --dk) {
885         int dkp = permutation[k + dk] - pk;
886         double gamma = 0;
887         for (int index = kDiag; index < jacobian.length; index += cols) {
888           gamma += jacobian[index] * jacobian[index + dkp];
889         }
890         gamma *= betak;
891         for (int index = kDiag; index < jacobian.length; index += cols) {
892           jacobian[index + dkp] -= gamma * jacobian[index];
893         }
894       }
895 
896     }
897 
898     rank = solvedCols;
899 
900   }

902   /*** 
903    * Compute the product Qt.y for some Q.R. decomposition.
904    * 
905    * @param y vector to multiply (will be overwritten with the result)
906    */
907   private void qTy(double[] y) {
908     for (int k = 0; k < cols; ++k) {
909       int pk = permutation[k];
910       int kDiag = k * cols + pk;
911       double gamma = 0;
912       for (int i = k, index = kDiag; i < rows; ++i, index += cols) {
913         gamma += jacobian[index] * y[i];
914       }
915       gamma *= beta[pk];
916       for (int i = k, index = kDiag; i < rows; ++i, index += cols) {
917         y[i] -= gamma * jacobian[index];
918       }
919     }
920   }
