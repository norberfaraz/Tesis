196   public void testTwoSets() throws EstimationException {
197     EstimatedParameter[] p = {
198       new EstimatedParameter("p0", 0),
199       new EstimatedParameter("p1", 1),
200       new EstimatedParameter("p2", 2),
201       new EstimatedParameter("p3", 3),
202       new EstimatedParameter("p4", 4),
203       new EstimatedParameter("p5", 5)
204     };
205 
206     double epsilon = 1.0e-7;
207     LinearProblem problem = new LinearProblem(new LinearMeasurement[] {
208 
209       // 4 elements sub-problem
210       new LinearMeasurement(new double[] {  2.0,  1.0,  4.0 },
211                             new EstimatedParameter[] { p[0], p[1], p[3] },
212                             2.0),
213       new LinearMeasurement(new double[] { -4.0, -2.0,   3.0, -7.0 },
214                            new EstimatedParameter[] { p[0], p[1], p[2], p[3] },
215                            -9.0),
216       new LinearMeasurement(new double[] {  4.0,  1.0,  -2.0,  8.0 },
217                             new EstimatedParameter[] { p[0], p[1], p[2], p[3] },
218                             2.0),
219       new LinearMeasurement(new double[] { -3.0, -12.0, -1.0 },
220                            new EstimatedParameter[] { p[1], p[2], p[3] },
221                            2.0),
222 
223       // 2 elements sub-problem
224       new LinearMeasurement(new double[] { epsilon, 1.0 },
225                             new EstimatedParameter[] { p[4], p[5] },
226                             1.0 + epsilon * epsilon),
227       new LinearMeasurement(new double[] {  1.0, 1.0 },
228                             new EstimatedParameter[] { p[4], p[5] },
229                             2.0)
230 
231     });
232 
233     LevenbergMarquardtEstimator estimator = new LevenbergMarquardtEstimator();
234     estimator.estimate(problem);
235     assertEquals(0, estimator.getRMS(problem), 1.0e-10);
236     assertEquals( 3.0, p[0].getEstimate(), 1.0e-10);
237     assertEquals( 4.0, p[1].getEstimate(), 1.0e-10);
238     assertEquals(-1.0, p[2].getEstimate(), 1.0e-10);
239     assertEquals(-2.0, p[3].getEstimate(), 1.0e-10);
240     assertEquals( 1.0 + epsilon, p[4].getEstimate(), 1.0e-10);
241     assertEquals( 1.0 - epsilon, p[5].getEstimate(), 1.0e-10);
242 
243   }

245   public void testNonInversible() throws EstimationException {
246 
247     EstimatedParameter[] p = {
248        new EstimatedParameter("p0", 0),
249        new EstimatedParameter("p1", 0),
250        new EstimatedParameter("p2", 0)
251     };
252     LinearMeasurement[] m = new LinearMeasurement[] {
253       new LinearMeasurement(new double[] {  1.0, 2.0, -3.0 },
254                             new EstimatedParameter[] { p[0], p[1], p[2] },
255                             1.0),
256       new LinearMeasurement(new double[] {  2.0, 1.0,  3.0 },
257                             new EstimatedParameter[] { p[0], p[1], p[2] },
258                             1.0),
259       new LinearMeasurement(new double[] { -3.0, -9.0 },
260                             new EstimatedParameter[] { p[0], p[2] },
261                             1.0)
262     };
263     LinearProblem problem = new LinearProblem(m);
264 
265     LevenbergMarquardtEstimator estimator = new LevenbergMarquardtEstimator();
266     double initialCost = estimator.getRMS(problem);
267     estimator.estimate(problem);
268     assertTrue(estimator.getRMS(problem) < initialCost);
269     assertTrue(Math.sqrt(m.length) * estimator.getRMS(problem) > 0.6);
270     double dJ0 = 2 * (m[0].getResidual() * m[0].getPartial(p[0])
271                     + m[1].getResidual() * m[1].getPartial(p[0])
272                     + m[2].getResidual() * m[2].getPartial(p[0]));
273     double dJ1 = 2 * (m[0].getResidual() * m[0].getPartial(p[1])
274                     + m[1].getResidual() * m[1].getPartial(p[1]));
275     double dJ2 = 2 * (m[0].getResidual() * m[0].getPartial(p[2])
276                     + m[1].getResidual() * m[1].getPartial(p[2])
277                     + m[2].getResidual() * m[2].getPartial(p[2]));
278     assertEquals(0, dJ0, 1.0e-10);
279     assertEquals(0, dJ1, 1.0e-10);
280     assertEquals(0, dJ2, 1.0e-10);
281 
282   }

639     private class PointModel extends WeightedMeasurement {
640 
641       public PointModel(double px, double py) {
642         super(1.0, 0.0);
643         this.px = px;
644         this.py = py;
645       }
646 
647       public double getPartial(EstimatedParameter parameter) {
648         if (parameter == cx) {
649           return getPartialDiX() - getPartialRadiusX();
650         } else if (parameter == cy) {
651           return getPartialDiY() - getPartialRadiusY();
652         }
653         return 0;
654       }
655 
656       public double getCenterDistance() {
657         double dx = px - cx.getEstimate();
658         double dy = py - cy.getEstimate();
659         return Math.sqrt(dx * dx + dy * dy);
660       }
661 
662       public double getPartialDiX() {
663         return (cx.getEstimate() - px) / getCenterDistance();
664       }
665 
666       public double getPartialDiY() {
667         return (cy.getEstimate() - py) / getCenterDistance();
668       }
669 
670       public double getTheoreticalValue() {
671         return getCenterDistance() - getRadius();
672       }
