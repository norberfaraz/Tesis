57      /*** 
58        * Generate 1000 random values and make sure they look OK.<br>
59        * Note that there is a non-zero (but very small) probability that
60        * these tests will fail even if the code is working as designed.
61        */
62      public void testNextDigest() throws Exception{
63          double next = 0.0;
64          double tolerance = 0.1;
65          vs.computeDistribution();
66          assertTrue("empirical distribution property", 
67              vs.getEmpiricalDistribution() != null);
68          SummaryStatistics stats = SummaryStatistics.newInstance();
69          for (int i = 1; i < 1000; i++) {
70              next = vs.getNext();
71              stats.addValue(next);
72          }    
73          assertEquals("mean", 5.069831575018909, stats.getMean(), tolerance);
74          assertEquals
75           ("std dev", 1.0173699343977738, stats.getStandardDeviation(), 
76              tolerance);
77          
78          vs.computeDistribution(500);
79          stats = SummaryStatistics.newInstance();
80          for (int i = 1; i < 1000; i++) {
81              next = vs.getNext();
82              stats.addValue(next);
83          }    
84          assertEquals("mean", 5.069831575018909, stats.getMean(), tolerance);
85          assertEquals
86           ("std dev", 1.0173699343977738, stats.getStandardDeviation(), 
87              tolerance);
88          
89      }

102     /***
103      * Test ValueServer REPLAY_MODE using values in testData file.<br> 
104      * Check that the values 1,2,1001,1002 match data file values 1 and 2.
105      * the sample data file.
106      */
107     public void testReplay() throws Exception {
108         double firstDataValue = 4.038625496201205;
109         double secondDataValue = 3.6485326248346936;
110         double tolerance = 10E-15;
111         double compareValue = 0.0d;
112         vs.setMode(ValueServer.REPLAY_MODE);
113         vs.resetReplayFile();
114         compareValue = vs.getNext();
115         assertEquals(compareValue,firstDataValue,tolerance);
116         compareValue = vs.getNext();
117         assertEquals(compareValue,secondDataValue,tolerance);
118         for (int i = 3; i < 1001; i++) {
119            compareValue = vs.getNext();
120         }
121         compareValue = vs.getNext();
122         assertEquals(compareValue,firstDataValue,tolerance);
123         compareValue = vs.getNext();
124         assertEquals(compareValue,secondDataValue,tolerance);
125         vs.closeReplayFile();
126         // make sure no NPE
127         vs.closeReplayFile();
128     }

130     /*** 
131      * Test other ValueServer modes
132      */
133     public void testModes() throws Exception {
134         vs.setMode(ValueServer.CONSTANT_MODE);
135         vs.setMu(0);
136         assertEquals("constant mode test",vs.getMu(),vs.getNext(),Double.MIN_VALUE);
137         vs.setMode(ValueServer.UNIFORM_MODE);
138         vs.setMu(2);
139         double val = vs.getNext();
140         assertTrue(val > 0 && val < 4);
141         vs.setSigma(1);
142         vs.setMode(ValueServer.GAUSSIAN_MODE);
143         val = vs.getNext();
144         assertTrue("gaussian value close enough to mean",
145             val < vs.getMu() + 100*vs.getSigma());
146         vs.setMode(ValueServer.EXPONENTIAL_MODE);
147         val = vs.getNext();
148         assertTrue(val > 0);
149         try {
150             vs.setMode(1000);
151             vs.getNext();
152             fail("bad mode, expecting IllegalStateException");
153         } catch (IllegalStateException ex) {
154             ;
155         }
156     }
