281     public void testPolar2Complex() {
282         TestUtils.assertEquals(Complex.ONE, 
283                 ComplexUtils.polar2Complex(1, 0), 10e-12);
284         TestUtils.assertEquals(Complex.ZERO, 
285                 ComplexUtils.polar2Complex(0, 1), 10e-12);
286         TestUtils.assertEquals(Complex.ZERO, 
287                 ComplexUtils.polar2Complex(0, -1), 10e-12);
288         TestUtils.assertEquals(Complex.I, 
289                 ComplexUtils.polar2Complex(1, pi/2), 10e-12);
290         TestUtils.assertEquals(Complex.I.negate(), 
291                 ComplexUtils.polar2Complex(1, -pi/2), 10e-12);
292         double r = 0;
293         for (int i = 0; i < 5; i++) {
294           r += i;
295           double theta = 0;
296           for (int j =0; j < 20; j++) {
297               theta += pi / 6;
298               TestUtils.assertEquals(altPolar(r, theta), 
299                       ComplexUtils.polar2Complex(r, theta), 10e-12);
300           }
301           theta = -2 * pi;
302           for (int j =0; j < 20; j++) {
303               theta -= pi / 6;
304               TestUtils.assertEquals(altPolar(r, theta), 
305                       ComplexUtils.polar2Complex(r, theta), 10e-12);
306           }
307         }   
308     }
309     
310     protected Complex altPolar(double r, double theta) {
311         return ComplexUtils.exp(Complex.I.multiply
312                 (new Complex(theta, 0))).multiply(new Complex(r, 0));
313     }
